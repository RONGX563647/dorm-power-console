    二、需求分析与设计¶

    2.1 业务场景¶

    场景 1：远程开关控制

    用户点击开关 → 前端调用 API → 生成命令 → MQTT 下发 → 设备执行 → 返回 ACK → 更新状态

    场景 2：批量控制

    用户选择多个设备 → 批量下发命令 → 命令队列处理 → 逐个执行 → 统计结果

    场景 3：定时任务

    用户设置定时任务 → 保存任务 → 定时检查 → 到点触发 → 下发命令 → 执行控制

    场景 4：场景联动

    触发条件满足 → 执行场景规则 → 批量下发命令 → 多个设备联动

    2.2 功能需求清单¶

    功能 描述 优先级
    远程控制 开关、参数设置 P0
    命令下发 MQTT 命令推送 P0
    命令确认 ACK 确认机制 P0
    命令队列 异步命令处理 P1
    批量控制 多设备同时控制 P1
    定时任务 定时执行控制 P2

    命令历史 记录查询 P2

    2.3 技术选型与理由¶

    技术 说明 版本 选型理由
    MQTT 命令下发协议 3.1.1 ✅ 双向通信：服务端发布命令，设备订阅并执行
    ✅ QoS 保障：关键命令使用 QoS 1 确保到达
    ✅ ACK 机制：设备执行后返回确认消息
    ✅ 主题隔离：不同设备使用不同主题（dorm/cmd/{deviceId}）
    Command Pattern 命令模式 - ✅ 封装请求：将命令封装为对象，便于存储和追踪
    ✅ 解耦调用者：发送者与接收者解耦
    ✅ 支持撤销：可扩展支持命令撤销功能
    ✅ 队列支持：命令可存入队列异步执行
    Quartz 定时任务调度 2.x ✅ Cron 表达式：灵活的定时策略
    ✅ 持久化：支持任务存储到数据库
    ✅ 集群支持：支持分布式任务调度
    ⚠️ 替代方案：教学场景用 Spring Scheduler 简化实现
    Redis 命令队列缓存 7.x ✅ 高性能：内存存储，适合高频命令队列
    ✅ List 结构：天然支持队列操作（LPUSH/RPOP）
    ✅ 持久化：支持 RDB/AOF 持久化
    ⚠️ 简化实现：教学场景直接用数据库存储命令

    技术对比与取舍： - 同步 vs 异步：选择异步命令队列，避免设备离线导致请求超时 - Redis vs 数据库：高频场景用 Redis，低频场景用数据库简化架构 - Quartz vs Spring Scheduler：简单定时用 Spring Scheduler，复杂调度用 Quartz

    package com.dormpower.model;

import jakarta.persistence.*;
import lombok.Data;
import java.time.LocalDateTime;
import java.util.Map;
import java.util.UUID;

@Data
@Entity
@Table(name = "device_command", indexes = {
    @Index(name = "idx_command_device_status", columnList = "deviceId, status"),
    @Index(name = "idx_command_created", columnList = "createdAt")
})
public class DeviceCommand {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(unique = true, nullable = false, length = 50)
    private String commandNumber; // 命令编号

    @Column(name = "device_id", nullable = false)
    private UUID deviceId;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private CommandType commandType;

    @Column(nullable = false)
    private Map<String, Object> parameters; // 命令参数

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private CommandStatus status = CommandStatus.PENDING;

    @Column(length = 500)
    private String mqttTopic; // MQTT 主题

    @Column(length = 1000)
    private String mqttPayload; // MQTT 消息体

    private Integer qos = 1; // MQTT QoS 级别

    private Integer timeoutSeconds = 30; // 超时时间

    private LocalDateTime sentAt; // 发送时间
    private LocalDateTime acknowledgedAt; // 确认时间
    private LocalDateTime executedAt; // 执行时间
    private LocalDateTime expiredAt; // 过期时间

    @Column(length = 500)
    private String errorMessage; // 错误信息

    private Integer retryCount = 0; // 重试次数

    @Column(name = "created_by")
    private UUID createdBy;

    private LocalDateTime createdAt = LocalDateTime.now();

    @PrePersist
    protected void onCreate() {
        if (commandNumber == null) {
            commandNumber = "CMD-" + LocalDateTime.now().format(
                java.time.format.DateTimeFormatter.ofPattern("yyyyMMddHHmmss")) +
                "-" + UUID.randomUUID().toString().substring(0, 4).toUpperCase();
        }
        if (expiredAt == null) {
            expiredAt = createdAt.plusMinutes(timeoutSeconds / 60);
        }
    }
}

enum CommandType {
    SWITCH_ON,          // 开
    SWITCH_OFF,         // 关
    RESTART,            // 重启
    SET_PARAMETER,      // 设置参数
    READ_STATUS,        // 读取状态
    FIRMWARE_UPDATE,    // 固件升级
    CUSTOM              // 自定义
}

enum CommandStatus {
    PENDING,      // 待发送
    SENT,         // 已发送
    ACKNOWLEDGED, // 已确认
    EXECUTED,     // 已执行
    FAILED,       // 失败
    TIMEOUT,      // 超时
    CANCELLED     // 已取消
}

    3.2 创建 CommandServiceImpl¶

    文件位置: backend/src/main/java/com/dormpower/service/impl/CommandServiceImpl.java

    package com.dormpower.service.impl;

import com.dormpower.dto.CommandRequest;
import com.dormpower.model.DeviceCommand;
import com.dormpower.model.CommandStatus;
import com.dormpower.model.CommandType;
import com.dormpower.repository.DeviceCommandRepository;
import com.dormpower.service.CommandService;
import com.dormpower.mqtt.MqttBridge;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;
import java.util.UUID;

@Slf4j
@Service
@RequiredArgsConstructor
public class CommandServiceImpl implements CommandService {

    private final DeviceCommandRepository commandRepository;
    private final MqttBridge mqttBridge;
    private final ObjectMapper objectMapper;

    @Override
    @Transactional
    public DeviceCommand sendCommand(CommandRequest request) {
        DeviceCommand command = new DeviceCommand();
        command.setDeviceId(request.getDeviceId());
        command.setCommandType(request.getCommandType());
        command.setParameters(request.getParameters());
        command.setQos(request.getQos() != null ? request.getQos() : 1);
        command.setTimeoutSeconds(request.getTimeoutSeconds() != null ? request.getTimeoutSeconds() : 30);

        // 构建 MQTT 消息
        String topic = "dorm/cmd/" + request.getDeviceId();
        command.setMqttTopic(topic);

        try {
            String payload = objectMapper.writeValueAsString(Map.of(
                "commandId", command.getCommandNumber(),
                "type", request.getCommandType().name(),
                "parameters", request.getParameters(),
                "timestamp", System.currentTimeMillis()
            ));
            command.setMqttPayload(payload);

            // 保存命令
            commandRepository.save(command);

            // 发送 MQTT 消息
            mqttBridge.publish(topic, payload, command.getQos());

            command.setStatus(CommandStatus.SENT);
            command.setSentAt(LocalDateTime.now());

            log.info("命令已发送：{}", command.getCommandNumber());
            return command;

        } catch (Exception e) {
            command.setStatus(CommandStatus.FAILED);
            command.setErrorMessage("发送失败：" + e.getMessage());
            log.error("命令发送失败：{}", command.getCommandNumber(), e);
            return commandRepository.save(command);
        }
    }

    @Override
    @Transactional
    public DeviceCommand acknowledgeCommand(String commandNumber, boolean success) {
        DeviceCommand command = commandRepository.findByCommandNumber(commandNumber)
            .orElseThrow(() -> new RuntimeException("命令不存在"));

        if (success) {
            command.setStatus(CommandStatus.ACKNOWLEDGED);
            command.setAcknowledgedAt(LocalDateTime.now());
        } else {
            command.setStatus(CommandStatus.FAILED);
            command.setErrorMessage("设备执行失败");
        }

        return commandRepository.save(command);
    }

    @Override
    public Page<DeviceCommand> getCommands(UUID deviceId, Pageable pageable) {
        return commandRepository.findByDeviceId(deviceId, pageable);
    }

    @Override
    public Optional<DeviceCommand> getCommand(String commandNumber) {
        return commandRepository.findByCommandNumber(commandNumber);
    }

    @Override
    @Transactional
    public void retryCommand(UUID commandId) {
        DeviceCommand command = commandRepository.findById(commandId)
            .orElseThrow(() -> new RuntimeException("命令不存在"));

        if (command.getRetryCount() >= 3) {
            throw new RuntimeException("已达到最大重试次数");
        }

        command.setRetryCount(command.getRetryCount() + 1);
        command.setStatus(CommandStatus.PENDING);
        command.setErrorMessage(null);

        // 重新发送
        mqttBridge.publish(command.getMqttTopic(), command.getMqttPayload(), command.getQos());
        command.setStatus(CommandStatus.SENT);
        command.setSentAt(LocalDateTime.now());

        commandRepository.save(command);
    }

    @Override
    @Transactional
    public void cancelCommand(UUID commandId) {
        DeviceCommand command = commandRepository.findById(commandId)
            .orElseThrow(() -> new RuntimeException("命令不存在"));

        if (command.getStatus() == CommandStatus.SENT || 
            command.getStatus() == CommandStatus.PENDING) {
            command.setStatus(CommandStatus.CANCELLED);
            commandRepository.save(command);
        } else {
            throw new RuntimeException("命令状态不允许取消");
        }
    }

    @Override
    @Transactional
    public List<DeviceCommand> sendBatchCommands(List<CommandRequest> requests) {
        return requests.stream()
            .map(this::sendCommand)
            .toList();
    }
}

    package com.dormpower.controller;

import com.dormpower.dto.CommandRequest;
import com.dormpower.model.DeviceCommand;
import com.dormpower.service.CommandService;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

import java.util.List;
import java.util.Map;
import java.util.UUID;

@RestController
@RequestMapping("/api/commands")
@RequiredArgsConstructor
public class CommandController {

    private final CommandService commandService;

    @PostMapping
    @PreAuthorize("hasAnyAuthority('device:control', 'admin')")
    public ResponseEntity<DeviceCommand> sendCommand(@RequestBody CommandRequest request) {
        return ResponseEntity.ok(commandService.sendCommand(request));
    }

    @PostMapping("/batch")
    @PreAuthorize("hasAnyAuthority('device:control', 'admin')")
    public ResponseEntity<List<DeviceCommand>> sendBatchCommands(
        @RequestBody List<CommandRequest> requests
    ) {
        return ResponseEntity.ok(commandService.sendBatchCommands(requests));
    }

    @GetMapping("/{commandNumber}")
    @PreAuthorize("hasAnyAuthority('device:read', 'admin')")
    public ResponseEntity<DeviceCommand> getCommand(@PathVariable String commandNumber) {
        return commandService.getCommand(commandNumber)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }

    @GetMapping("/device/{deviceId}")
    @PreAuthorize("hasAnyAuthority('device:read', 'admin')")
    public ResponseEntity<Page<DeviceCommand>> getDeviceCommands(
        @PathVariable UUID deviceId,
        Pageable pageable
    ) {
        return ResponseEntity.ok(commandService.getCommands(deviceId, pageable));
    }

    @PostMapping("/{id}/acknowledge")
    @PreAuthorize("hasAnyAuthority('device:control', 'admin')")
    public ResponseEntity<DeviceCommand> acknowledgeCommand(
        @PathVariable UUID id,
        @RequestParam boolean success
    ) {
        DeviceCommand command = commandRepository.findById(id)
            .orElseThrow(() -> new RuntimeException("命令不存在"));
        return ResponseEntity.ok(commandService.acknowledgeCommand(command.getCommandNumber(), success));
    }

    @PostMapping("/{id}/retry")
    @PreAuthorize("hasAnyAuthority('device:control', 'admin')")
    public ResponseEntity<Void> retryCommand(@PathVariable UUID id) {
        commandService.retryCommand(id);
        return ResponseEntity.ok().build();
    }

    @PostMapping("/{id}/cancel")
    @PreAuthorize("hasAnyAuthority('device:control', 'admin')")
    public ResponseEntity<Void> cancelCommand(@PathVariable UUID id) {
        commandService.cancelCommand(id);
        return ResponseEntity.ok().build();
    }
}

    import axios from 'axios';
import type { DeviceCommand, CommandRequest, ScheduledTask } from '@/types/command';

const API_BASE = '/api/commands';

export const commandApi = {
  // 发送命令
  async sendCommand(request: CommandRequest): Promise<DeviceCommand> {
    const response = await axios.post(API_BASE, request);
    return response.data;
  },

  // 批量发送命令
  async sendBatchCommands(requests: CommandRequest[]): Promise<DeviceCommand[]> {
    const response = await axios.post(`${API_BASE}/batch`, requests);
    return response.data;
  },

  // 获取命令详情
  async getCommand(commandNumber: string): Promise<DeviceCommand> {
    const response = await axios.get(`${API_BASE}/${commandNumber}`);
    return response.data;
  },

  // 获取设备命令列表
  async getDeviceCommands(deviceId: string, page = 0, size = 20) {
    const response = await axios.get(`${API_BASE}/device/${deviceId}`, {
      params: { page, size }
    });
    return response.data;
  },

  // 确认命令
  async acknowledgeCommand(commandId: string, success: boolean): Promise<DeviceCommand> {
    const response = await axios.post(`${API_BASE}/${commandId}/acknowledge`, null, {
      params: { success }
    });
    return response.data;
  },

  // 重试命令
  async retryCommand(commandId: string) {
    await axios.post(`${API_BASE}/${commandId}/retry`);
  },

  // 取消命令
  async cancelCommand(commandId: string) {
    await axios.post(`${API_BASE}/${commandId}/cancel`);
  }
};

// 定时任务 API
const TASK_API_BASE = '/api/scheduled-tasks';

export const taskApi = {
  // 获取任务列表
  async getTasks(enabled?: boolean) {
    const response = await axios.get(TASK_API_BASE, { params: { enabled } });
    return response.data;
  },

  // 创建任务
  async createTask(task: Partial<ScheduledTask>) {
    const response = await axios.post(TASK_API_BASE, task);

  },

import { ref, computed, onMounted } from 'vue';

const devices = ref<any[]>([]);

const deviceStatus = ref(false);

const filteredDevices = computed(() => {
  if (!searchText.value) return devices.value;
  return devices.value.filter(d =>

  );

const loadDevices = async () => {
  try {
    const response = await deviceApi.getDevices();

  }

const handleDeviceSelect = (device: any) => {
  selectedDevice.value = device;
  deviceStatus.value = device.status === 'ONLINE';

const handleSwitchChange = async (value: boolean) => {
  if (!selectedDevice.value) return;

  try {
    await commandApi.sendCommand({
      deviceId: selectedDevice.value.id,
      commandType: value ? 'SWITCH_ON' : 'SWITCH_OFF',
      parameters: {}
    });

    ElMessage.success('命令已发送');
  } catch (error: any) {
    ElMessage.error('发送失败：' + (error.response?.data?.message || ''));
  }
};

// 重启设备
const handleRestart = async () => {
  if (!selectedDevice.value) return;

  try {
    await commandApi.sendCommand({
      deviceId: selectedDevice.value.id,
      commandType: 'RESTART',
      parameters: {}
    });

    ElMessage.success('重启命令已发送');
  } catch (error: any) {

  }

    问题 1：命令超时¶

    现象: 设备未及时响应

    解决方案:

    // 定时检查超时命令
@Scheduled(fixedRate = 60000) // 每分钟检查
public void checkTimeoutCommands() {
    List<DeviceCommand> timeoutCommands = commandRepository
        .findByStatusAndExpiredAtBefore(CommandStatus.SENT, LocalDateTime.now());

    for (DeviceCommand command : timeoutCommands) {
        command.setStatus(CommandStatus.TIMEOUT);
        command.setErrorMessage("命令超时");
        commandRepository.save(command);
    }
}
