# 控制管理模块 - 功能设计与实现细节文档

## 1. 模块概述

### 1.1 模块定位

控制管理模块是宿舍电源管理系统的核心模块之一，负责实现对设备的远程控制命令下发和状态跟踪。该模块提供命令下发、状态查询、历史记录、冲突检测和超时处理等功能，是用户与设备交互的关键桥梁。

#### 2.1.7 实现代码

**Controller层：**

```java
@RestController
@RequestMapping("/api")
@Tag(name = "设备控制", description = "提供设备控制命令下发的RESTful API接口")
public class CommandController {

    @Autowired
    private CommandService commandService;

    @Autowired
    private MqttBridge mqttBridge;

    @Operation(summary = "下发控制命令", description = "向指定设备下发控制命令")
    @RateLimit(value = 5.0, type = "device")
    @AuditLog(value = "下发设备控制命令", type = "CONTROL")
    @PostMapping("/strips/{deviceId}/cmd")
    public ResponseEntity<Map<String, Object>> sendCommand(
            @PathVariable String deviceId,
            @RequestBody Map<String, Object> request) {
        
        Map<String, Object> response = commandService.sendCommand(deviceId, request);
        
        String cmdId = (String) response.get("cmdId");
        Map<String, Object> payload = new HashMap<>(request);
        payload.put("cmdId", cmdId);

        boolean published = mqttBridge.publishCommand(deviceId, payload);
        if (!published) {
            commandService.updateCommandState(cmdId, "failed", "mqtt unavailable");
        }

        return ResponseEntity.ok(response);
    }

    @Operation(summary = "批量下发命令", description = "向多个设备同时下发相同的控制命令")
    @RateLimit(value = 2.0, type = "batch-command")
    @AuditLog(value = "批量下发设备控制命令", type = "CONTROL")
    @PostMapping("/commands/batch")
    public ResponseEntity<Map<String, Object>> sendBatchCommand(
            @RequestBody BatchCommandRequest request) {
        
        Map<String, Object> response = commandService.sendBatchCommand(
            request.deviceIds(), request.command());

        Map<String, String> commandIds = (Map<String, String>) response.get("commandIds");
        if (commandIds != null) {
            for (Map.Entry<String, String> entry : commandIds.entrySet()) {
                String deviceId = entry.getKey();
                String cmdId = entry.getValue();
                Map<String, Object> payload = new HashMap<>(request.command());
                payload.put("cmdId", cmdId);
                mqttBridge.publishCommand(deviceId, payload);
            }
        }

        return ResponseEntity.ok(response);
    }
}
```

**Service层：**

```java
@Service
public class CommandService {

    private static final int CMD_TIMEOUT_SECONDS = 30;
    private static final String STATE_PENDING = "pending";
    private static final String STATE_SUCCESS = "success";
    private static final String STATE_FAILED = "failed";
    private static final String STATE_TIMEOUT = "timeout";

    @Autowired
    private CommandRecordRepository commandRecordRepository;

    @Autowired
    private DeviceRepository deviceRepository;

    public CommandRecord createCommandRecord(String deviceId, Map<String, Object> request) {
        final long now = System.currentTimeMillis();
        final String cmdId = "cmd-" + now;

        CommandRecord commandRecord = new CommandRecord();
        commandRecord.setCmdId(cmdId);
        commandRecord.setDeviceId(deviceId);
        commandRecord.setAction((String) request.getOrDefault("action", "toggle"));
        
        Object socketObj = request.get("socket");
        Integer socket = socketObj != null ? ((Number) socketObj).intValue() : null;
        commandRecord.setSocket(socket);

        try {
            commandRecord.setPayloadJson(objectMapper.writeValueAsString(request));
        } catch (Exception e) {
            commandRecord.setPayloadJson("{}");
        }

        commandRecord.setState(STATE_PENDING);
        commandRecord.setMessage("");
        commandRecord.setCreatedAt(now);
        commandRecord.setUpdatedAt(now);
        commandRecord.setExpiresAt(now + CMD_TIMEOUT_SECONDS * 1000L);

        return commandRecordRepository.save(commandRecord);
    }

    public Map<String, Object> sendCommand(String deviceId, Map<String, Object> request) {
        Device device = deviceRepository.findById(deviceId).orElse(null);
        if (device == null) {
            throw new ResourceNotFoundException("Device not found");
        }

        Object socketObj = request.get("socket");
        Integer socket = socketObj != null ? ((Number) socketObj).intValue() : null;
        
        if (hasPendingConflict(deviceId, socket)) {
            throw new BusinessException("Pending command exists for target");
        }

        CommandRecord commandRecord = createCommandRecord(deviceId, request);

        Map<String, Object> response = new HashMap<>(4);
        response.put("ok", true);
        response.put("cmdId", commandRecord.getCmdId());
        response.put("stripId", deviceId);
        response.put("acceptedAt", commandRecord.getCreatedAt());
        return response;
    }
}
```

```java
@RestController
@RequestMapping("/api")
@Tag(name = "命令查询", description = "设备控制命令状态查询接口")
public class CommandQueryController {

    @Autowired
    private CommandService commandService;

    @Operation(summary = "查询命令状态", description = "根据命令ID查询设备控制命令的执行状态")
    @GetMapping("/cmd/{cmdId}")
    public ResponseEntity<Map<String, Object>> getCommandStatus(
            @PathVariable String cmdId) {
        
        Map<String, Object> status = commandService.getCommandStatus(cmdId);
        if (status == null) {
            Map<String, Object> error = new HashMap<>();
            error.put("ok", false);
            error.put("code", "NOT_FOUND");
            error.put("message", "cmd not found");
            return ResponseEntity.status(HttpStatus.NOT_FOUND).body(error);
        }
        return ResponseEntity.ok(status);
    }
}
```

```java
@Service
public class CommandService {

    public Map<String, Object> getCommandStatus(String cmdId) {
        CommandRecord commandRecord = commandRecordRepository.findById(cmdId).orElse(null);
        
        if (commandRecord == null) {
            return null;
        }

        Map<String, Object> status = new HashMap<>(4);
        status.put("cmdId", commandRecord.getCmdId());
        status.put("state", commandRecord.getState());
        status.put("message", commandRecord.getMessage());
        status.put("updatedAt", commandRecord.getUpdatedAt());
        
        Integer duration = commandRecord.getDurationMs();
        if (duration != null) {
            status.put("durationMs", duration);
        }
        
        return status;
    }
}
```

```java
@Service
public class CommandService {

    public boolean hasPendingConflict(String deviceId, Integer socket) {
        List<CommandRecord> pendingCommands = commandRecordRepository
            .findByDeviceIdAndState(deviceId, STATE_PENDING);
        
        for (CommandRecord cmd : pendingCommands) {
            Integer cmdSocket = cmd.getSocket();
            
            // 两个socket都为null，表示都是全局命令，冲突
            if (socket == null && cmdSocket == null) {
                return true;
            }
            
            // 两个socket相同，冲突
            if (socket != null && socket.equals(cmdSocket)) {
                return true;
            }
        }
        
        return false;
    }

    public Map<String, Object> sendCommand(String deviceId, Map<String, Object> request) {
        Device device = deviceRepository.findById(deviceId).orElse(null);
        if (device == null) {
            throw new ResourceNotFoundException("Device not found");
        }

        Object socketObj = request.get("socket");
        Integer socket = socketObj != null ? ((Number) socketObj).intValue() : null;
        
        // 冲突检测
        if (hasPendingConflict(deviceId, socket)) {
            throw new BusinessException("Pending command exists for target");
        }

        // 创建命令记录...
    }
}
```

```java
@Service
public class CommandService {

    private static final int CMD_TIMEOUT_SECONDS = 30;
    private static final String STATE_PENDING = "pending";
    private static final String STATE_TIMEOUT = "timeout";

    @Autowired
    private CommandRecordRepository commandRecordRepository;

    /**
     * 标记超时命令
     * 由定时任务调用，定期检查pending状态的命令是否超时
     */
    public void markTimeouts() {
        final long now = System.currentTimeMillis();
        
        // 查询所有pending状态的命令
        List<CommandRecord> pendingCommands = commandRecordRepository
            .findByState(STATE_PENDING);
        
        for (CommandRecord cmd : pendingCommands) {
            // 检查是否超时
            if (cmd.getExpiresAt() < now) {
                cmd.setState(STATE_TIMEOUT);
                cmd.setMessage("Command timed out");
                cmd.setUpdatedAt(now);
                commandRecordRepository.save(cmd);
                
                logger.warn("命令超时: cmdId={}, deviceId={}", 
                    cmd.getCmdId(), cmd.getDeviceId());
            }
        }
    }
}
```

### 5.1 MQTT命令下发

```java
@Component
public class MqttBridge {

    @Autowired
    private MqttConfig mqttConfig;

    @Autowired
    private ObjectMapper objectMapper;

    private MqttClient mqttClient;
    private boolean connected = false;

    /**
     * 发布命令到设备
     * 
     * @param deviceId 设备ID
     * @param payload 命令负载
     * @return 是否发布成功
     */
    public boolean publishCommand(String deviceId, Object payload) {
        if (!connected || mqttClient == null || !mqttClient.isConnected()) {
            logger.error("MQTT未连接，无法下发命令");
            return false;
        }

        try {
            // 构建命令主题
            String topic = mqttConfig.buildTopic(deviceId, "cmd");
            
            // 序列化命令负载
            String jsonPayload = objectMapper.writeValueAsString(payload);
            
            // 创建MQTT消息，QoS=1保证至少一次送达
            MqttMessage message = new MqttMessage(jsonPayload.getBytes());
            message.setQos(1);
            
            // 发布消息
            mqttClient.publish(topic, message);
            
            logger.info("命令已发布: topic={}, payload={}", topic, jsonPayload);
            return true;
        } catch (Exception e) {
            logger.error("发布命令失败: {}", e.getMessage());
            return false;
        }
    }
}
```
