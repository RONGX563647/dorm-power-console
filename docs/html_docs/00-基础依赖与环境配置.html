一、项目概述

1.1 技术栈总览

后端技术栈:
  核心框架:
    - Java 21 (LTS)
    - Spring Boot 3.2.3
    - Spring Security (认证授权)
    - Spring Data JPA (数据持久化)

  通信协议:
    - MQTT (IoT设备通信)
    - WebSocket (实时推送)
    - REST API (前后端交互)

  数据存储:
    - PostgreSQL (主数据库)
    - H2 (开发环境内存数据库)
    - Redis (分布式缓存)
    - Caffeine (本地缓存)

  消息中间件:
    - RabbitMQ (消息队列)
    - Kafka (高吞吐消息处理)

  监控与可观测性:
    - Spring Boot Actuator
    - Prometheus + Grafana
    - Micrometer

  其他工具:
    - Lombok (代码简化)
    - JWT (Token认证)
    - Springdoc OpenAPI (API文档)
    - Guava (工具库)
    - Resilience4j (熔断降级)

前端技术栈:
  核心框架:
    - Vue 3.5 (Composition API)
    - TypeScript 5.7
    - Vite 6.2 (构建工具)

  UI组件库:
    - Ant Design Vue 4.2

  状态管理:
    - Pinia 3.0

  路由管理:
    - Vue Router 4.5

  数据可视化:
    - ECharts 5.6
    - Vue-ECharts 7.0

  HTTP客户端:
    - Axios 1.7

  工具库:
    - Day.js (时间处理)

  代码质量:
    - ESLint 9.2
    - Husky (Git钩子)
    - Commitlint (提交规范)

3.2 Vite 配置

文件位置: frontend/vite.config.ts

import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import { resolve } from 'path'
import { visualizer } from 'rollup-plugin-visualizer'

export default defineConfig({
  plugins: [
    vue(),
    // 打包分析（可选）
    process.env.ANALYZE && visualizer({
      open: true,
      gzipSize: true,
      brotliSize: true
    })
  ],

  resolve: {
    alias: {
      '@': resolve(__dirname, 'src')  // 路径别名
    }
  },

  server: {
    port: 3000,                        // 开发服务器端口
    proxy: {
      // API代理配置
      '/api': {
        target: 'http://localhost:8000',
        changeOrigin: true
      },
      '/ws': {
        target: 'http://localhost:8000',
        ws: true
      }
    }
  },

  build: {
    target: 'es2015',                  // 编译目标
    outDir: 'dist',                    // 输出目录
    sourcemap: false,                  // 生产环境不生成sourcemap
    rollupOptions: {
      output: {
        // 手动分包
        manualChunks: {
          vendor: ['vue', 'vue-router', 'pinia'],
          ui: ['ant-design-vue', '@ant-design/icons-vue'],
          charts: ['echarts', 'vue-echarts']
        }
      }
    },
    // 压缩配置
    minify: 'terser',
    terserOptions: {
      compress: {
        drop_console: true,            // 移除console
        drop_debugger: true            // 移除debugger
      }
    }
  }
})

5.2 SecurityConfig 安全配置

文件位置: backend/src/main/java/com/dormpower/config/SecurityConfig.java

package com.dormpower.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.config.http.SessionCreationPolicy;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.SecurityFilterChain;
import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter;
import lombok.RequiredArgsConstructor;

/**
 * Spring Security配置类
 * 
 * 配置说明:
 * 1. 禁用CSRF（前后端分离架构，使用JWT）
 * 2. 禁用Session（无状态认证）
 * 3. 配置URL访问权限
 * 4. 添加JWT过滤器
 */
@Configuration
@EnableWebSecurity
@RequiredArgsConstructor
public class SecurityConfig {

    private final JwtAuthenticationFilter jwtAuthenticationFilter;

    /**
     * 配置安全过滤链
     */
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            // 禁用CSRF（前后端分离）
            .csrf(csrf -> csrf.disable())

            // 禁用Session（JWT无状态）
            .sessionManagement(session -> 
                session.sessionCreationPolicy(SessionCreationPolicy.STATELESS))

            // 配置URL访问权限
            .authorizeHttpRequests(auth -> auth
                // 公开接口
                .requestMatchers(
                    "/api/auth/login",
                    "/api/auth/register",
                    "/api/auth/forgot-password",
                    "/h2-console/**",
                    "/swagger-ui/**",
                    "/v3/api-docs/**",
                    "/actuator/**",
                    "/ws/**"
                ).permitAll()

                // 其他接口需要认证
                .anyRequest().authenticated()
            )

            // 添加JWT过滤器
            .addFilterBefore(
                jwtAuthenticationFilter,
                UsernamePasswordAuthenticationFilter.class
            );

        return http.build();
    }

    /**
     * 密码加密器
     */
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder();
    }
}

5.3 CorsConfig 跨域配置

文件位置: backend/src/main/java/com/dormpower/config/CorsConfig.java

package com.dormpower.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.UrlBasedCorsConfigurationSource;
import org.springframework.web.filter.CorsFilter;

import java.util.List;

/**
 * CORS跨域配置
 * 
 * 配置说明:
 * 1. 允许前端域名访问
 * 2. 允许的HTTP方法
 * 3. 允许的请求头
 * 4. 凭证支持
 */
@Configuration
public class CorsConfig {

    @Bean
    public CorsFilter corsFilter() {
        CorsConfiguration config = new CorsConfiguration();

        // 允许的源（开发环境允许localhost:3000）
        config.setAllowedOrigins(List.of(
            "http://localhost:3000",
            "http://localhost:5173"
        ));

        // 允许凭证（Cookie、Authorization头）
        config.setAllowCredentials(true);

        // 允许的HTTP方法
        config.setAllowedMethods(List.of(
            "GET", "POST", "PUT", "DELETE", "OPTIONS"
        ));

        // 允许的请求头
        config.setAllowedHeaders(List.of("*"));

        // 预检请求缓存时间（秒）
        config.setMaxAge(3600L);

        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
        source.registerCorsConfiguration("/**", config);

        return new CorsFilter(source);
    }
}

5.4 AsyncConfig 异步配置

文件位置: backend/src/main/java/com/dormpower/config/AsyncConfig.java

package com.dormpower.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.annotation.EnableAsync;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

import java.util.concurrent.Executor;

/**
 * 异步任务配置
 * 
 * 配置说明:
 * 1. 启用异步支持（@EnableAsync）
 * 2. 配置线程池参数
 * 3. 自定义线程名称
 */
@Configuration
@EnableAsync
public class AsyncConfig {

    @Bean(name = "taskExecutor")
    public Executor taskExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();

        // 核心线程数
        executor.setCorePoolSize(4);

        // 最大线程数
        executor.setMaxPoolSize(8);

        // 队列容量
        executor.setQueueCapacity(100);

        // 线程名称前缀
        executor.setThreadNamePrefix("AsyncTask-");

        // 等待所有任务结束后再关闭线程池
        executor.setWaitForTasksToCompleteOnShutdown(true);

        // 等待时间（秒）
        executor.setAwaitTerminationSeconds(60);

        // 拒绝策略：由调用线程处理
        executor.setRejectedExecutionHandler(
            new java.util.concurrent.ThreadPoolExecutor.CallerRunsPolicy()
        );

        executor.initialize();
        return executor;
    }
}

5.10 RateLimitConfig 限流配置

package com.dormpower.config;

import com.google.common.util.concurrent.RateLimiter;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 限流配置（基于Guava RateLimiter）
 * 
 * 配置说明:
 * 1. 令牌桶算法限流
 * 2. 支持不同接口不同限流策略
 */
@Configuration
public class RateLimitConfig {

    // 存储各接口的限流器
    private final Map<String, RateLimiter> rateLimiters = new ConcurrentHashMap<>();

    /**
     * 获取或创建限流器
     * 
     * @param key 限流键
     * @param permitsPerSecond 每秒许可数
     */
    public RateLimiter getRateLimiter(String key, double permitsPerSecond) {
        return rateLimiters.computeIfAbsent(key, k -> 
            RateLimiter.create(permitsPerSecond)
        );
    }
}
