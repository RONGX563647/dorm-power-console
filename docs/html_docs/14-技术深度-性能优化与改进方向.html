@Component
public class PerformanceMonitor {

    private final MeterRegistry meterRegistry;
    private final Timer requestTimer;

    public PerformanceMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        this.requestTimer = Timer.builder("http.request.duration")
            .tag("application", "dorm-power")
            .register(meterRegistry);
    }

    public <T> T monitor(Supplier<T> operation, String operationName) {
        return requestTimer.record(() -> {
            long startTime = System.currentTimeMillis();
            try {
                return operation.get();
            } finally {
                long duration = System.currentTimeMillis() - startTime;
                if (duration > 1000) {
                    log.warn("Slow operation detected: {} took {}ms", 
                        operationName, duration);
                }
            }
        });
    }
}

2.1.2 参数说明

内存分配策略:
  堆内存:
    -Xms256m: "初始堆大小256MB（避免启动时频繁GC）"
    -Xmx384m: "最大堆大小384MB（预留系统内存）"

  元空间:
    -XX:MetaspaceSize=64m: "初始元空间64MB"
    -XX:MaxMetaspaceSize=128m: "最大元空间128MB"

  GC策略:
    -XX:+UseG1GC: "使用G1垃圾回收器（适合大堆低延迟）"
    -XX:MaxGCPauseMillis=200: "目标GC停顿时间200ms"

  诊断:
    -XX:+HeapDumpOnOutOfMemoryError: "OOM时自动dump堆"
    -Xlog:gc*: "输出详细GC日志"

     * 1. 观察GC频率
     *    - Young GC: 理想频率 10-50次/小时
     *    - Full GC: 理想频率 <1次/天

2.3 内存泄漏排查

2.3.1 常见内存泄漏场景

// 错误示例1: 静态集合持续增长
public class BadCacheExample {
    // 内存泄漏！无界集合
    private static final Map<String, byte[]> CACHE = new HashMap<>();

    public void addToCache(String key, byte[] data) {
        CACHE.put(key, data);  // 永远不会移除
    }
}

// 正确示例: 使用Caffeine有界缓存
public class GoodCacheExample {
    private static final Cache<String, byte[]> CACHE = Caffeine.newBuilder()
        .maximumSize(10000)  // 最大10000条
        .expireAfterWrite(Duration.ofMinutes(30))  // 30分钟过期
        .recordStats()
        .build();

    public void addToCache(String key, byte[] data) {
        CACHE.put(key, data);  // 自动淘汰旧数据
    }
}

// 错误示例2: 未关闭的资源
public class ResourceLeakExample {

    // 内存泄漏！Connection未关闭
    public List<TelemetryData> queryData() throws SQLException {
        Connection conn = dataSource.getConnection();
        PreparedStatement stmt = conn.prepareStatement("SELECT * FROM telemetry");
        ResultSet rs = stmt.executeQuery();

        List<TelemetryData> result = new ArrayList<>();
        while (rs.next()) {
            result.add(mapRow(rs));
        }
        // 忘记关闭rs, stmt, conn
        return result;
    }

    // 正确示例: try-with-resources
    public List<TelemetryData> queryDataFixed() throws SQLException {
        List<TelemetryData> result = new ArrayList<>();

        try (Connection conn = dataSource.getConnection();
             PreparedStatement stmt = conn.prepareStatement("SELECT * FROM telemetry");
             ResultSet rs = stmt.executeQuery()) {

            while (rs.next()) {
                result.add(mapRow(rs));
            }
        }

        return result;
    }
}

3.2.3 批量操作优化

@Service
@RequiredArgsConstructor
public class BatchOptimizationService {

    private final JdbcTemplate jdbcTemplate;

    // 批量插入优化
    @Transactional
    public void batchInsertTelemetry(List<TelemetryData> dataList) {
        String sql = "INSERT INTO telemetry (device_id, voltage, current, power, collected_at) " +
                    "VALUES (?, ?, ?, ?, ?)";

        jdbcTemplate.batchUpdate(sql, new BatchPreparedStatementSetter() {
            @Override
            public void setValues(PreparedStatement ps, int i) throws SQLException {
                TelemetryData data = dataList.get(i);
                ps.setString(1, data.getDeviceId());
                ps.setDouble(2, data.getVoltage());
                ps.setDouble(3, data.getCurrent());
                ps.setDouble(4, data.getPower());
                ps.setTimestamp(5, Timestamp.valueOf(data.getCollectedAt()));
            }

            @Override
            public int getBatchSize() {
                return dataList.size();
            }
        });
    }

    // 分批处理（避免内存溢出）
    public void processLargeDataset(int totalRecords, int batchSize, Consumer<List<Integer>> processor) {
        int offset = 0;
        while (offset < totalRecords) {
            List<Integer> batch = IntStream.range(offset, Math.min(offset + batchSize, totalRecords))
                .boxed()
                .collect(Collectors.toList());

            processor.accept(batch);
            offset += batchSize;
        }
    }
}

4.1.1 对象池复用

@Service
public class ObjectPoolService {

    // 使用对象池复用DateFormatter（线程不安全）
    private final ObjectPool<SimpleDateFormat> dateFormatterPool = new GenericObjectPool<>(
        new BasePooledObjectFactory<SimpleDateFormat>() {
            @Override
            public SimpleDateFormat create() {
                return new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
            }

            @Override
            public PooledObject<SimpleDateFormat> wrap(SimpleDateFormat obj) {
                return new DefaultPooledObject<>(obj);
            }
        }
    );

    public String formatDate(Date date) throws Exception {
        SimpleDateFormat sdf = null;
        try {
            sdf = dateFormatterPool.borrowObject();
            return sdf.format(date);
        } finally {
            if (sdf != null) {
                dateFormatterPool.returnObject(sdf);
            }
        }
    }
}

4.1.2 字符串优化

public class StringOptimization {

    // 错误: 循环中字符串拼接
    public String buildMessageBad(List<String> parts) {
        String result = "";
        for (String part : parts) {
            result += part;  // 每次创建新String对象
        }
        return result;
    }

    // 正确: 使用StringBuilder
    public String buildMessageGood(List<String> parts) {
        StringBuilder sb = new StringBuilder(parts.size() * 20);  // 预估容量
        for (String part : parts) {
            sb.append(part);
        }
        return sb.toString();
    }

    // 最佳: Stream API（简洁但稍慢）
    public String buildMessageStream(List<String> parts) {
        return parts.stream().collect(Collectors.joining());
    }
}

public class CollectionInitialization {

    // 错误: 默认容量，频繁扩容
    public List<TelemetryData> loadAllBad() {
        List<TelemetryData> list = new ArrayList<>();  // 初始容量10
        // 加载10000条数据，会扩容10+次
        return list;
    }

    // 正确: 预估容量
    public List<TelemetryData> loadAllGood(int expectedSize) {
        // 初始容量 = expectedSize / 0.75 + 1
        List<TelemetryData> list = new ArrayList<>((int)(expectedSize / 0.75f) + 1);
        return list;
    }

    // HashMap初始化
    public Map<String, Device> buildDeviceMap(int expectedSize) {
        // 避免扩容: capacity = expectedSize / loadFactor + 1
        Map<String, Device> map = new HashMap<>((int)(expectedSize / 0.75f) + 1);
        return map;
    }
}

4.3.1 锁优化

@Service
public class LockOptimization {

    // 错误: 粗粒度锁
    public synchronized void processDevicesBad(List<Device> devices) {
        // 整个方法加锁，串行执行
        for (Device device : devices) {
            processDevice(device);
        }
    }

    // 正确: 细粒度锁
    public void processDevicesGood(List<Device> devices) {
        for (Device device : devices) {
            synchronized (device) {  // 只锁单个设备
                processDevice(device);
            }
        }
    }

    // 最佳: 并发集合
    private final ConcurrentHashMap<String, AtomicInteger> deviceCounters = 
        new ConcurrentHashMap<>();

    public void incrementCounter(String deviceId) {
        deviceCounters.computeIfAbsent(deviceId, k -> new AtomicInteger(0))
            .incrementAndGet();
    }
}

@Service
@RequiredArgsConstructor
public class VirtualThreadService {

    private final ExecutorService virtualThreadExecutor;

    // 使用虚拟线程处理大量I/O任务
    public CompletableFuture<List<DeviceStatus>> checkAllDevicesAsync(List<String> deviceIds) {
        List<CompletableFuture<DeviceStatus>> futures = deviceIds.stream()
            .map(deviceId -> CompletableFuture.supplyAsync(
                () -> checkDeviceStatus(deviceId),
                virtualThreadExecutor
            ))
            .collect(Collectors.toList());

        return CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
            .thenApply(v -> futures.stream()
                .map(CompletableFuture::join)
                .collect(Collectors.toList()));
    }
}

5.1.2 缓存优化

// 改进: 多级缓存策略
@Service
public class MultiLevelCacheService {

    private final Cache<String, Device> localCache;  // Caffeine L1
    private final RedisTemplate<String, Device> redisTemplate;  // Redis L2

    public Device getDevice(String deviceId) {
        // L1: 本地缓存
        Device device = localCache.getIfPresent(deviceId);
        if (device != null) {
            return device;
        }

        // L2: Redis缓存
        device = redisTemplate.opsForValue().get("device:" + deviceId);
        if (device != null) {
            localCache.put(deviceId, device);  // 回填L1
            return device;
        }

        // L3: 数据库
        device = deviceRepository.findById(deviceId).orElse(null);
        if (device != null) {
            redisTemplate.opsForValue().set("device:" + deviceId, device, 
                Duration.ofMinutes(30));
            localCache.put(deviceId, device);
        }

        return device;
    }

    // 缓存更新策略: Cache-Aside
    public void updateDevice(Device device) {
        deviceRepository.save(device);
        localCache.invalidate(device.getDeviceId());
        redisTemplate.delete("device:" + device.getDeviceId());
    }
}

// 配置读写分离
@Configuration
public class DataSourceConfig {

    @Bean
    @ConfigurationProperties("spring.datasource.write")
    public DataSource writeDataSource() {
        return DataSourceBuilder.create().build();
    }

    @Bean
    @ConfigurationProperties("spring.datasource.read")
    public DataSource readDataSource() {
        return DataSourceBuilder.create().build();
    }

    @Bean
    public DataSource routingDataSource() {
        Map<Object, Object> targetDataSources = new HashMap<>();
        targetDataSources.put("write", writeDataSource());
        targetDataSources.put("read", readDataSource());

        RoutingDataSource routingDataSource = new RoutingDataSource();
        routingDataSource.setDefaultTargetDataSource(writeDataSource());
        routingDataSource.setTargetDataSources(targetDataSources);
        return routingDataSource;
    }
}

# 改进: AI异常检测
class AnomalyDetector:
    """基于机器学习的异常检测"""

    def __init__(self):
        self.model = self.load_model()

    def detect_anomaly(self, telemetry_data: List[TelemetryData]) -> AnomalyResult:
        """检测用电异常"""
        # 特征工程
        features = self.extract_features(telemetry_data)

        # 模型预测
        prediction = self.model.predict(features)

        # 生成报告
        return AnomalyResult(
            is_anomaly=prediction['is_anomaly'],
            confidence=prediction['confidence'],
            anomaly_type=prediction['type'],
            suggestions=self.generate_suggestions(prediction)
        )

    def extract_features(self, data: List[TelemetryData]) -> np.ndarray:
        """提取特征: 电压波动、电流峰值、功率趋势等"""
        voltages = [d.voltage for d in data]
        currents = [d.current for d in data]
        powers = [d.power for d in data]

        return np.array([
            np.std(voltages),  # 电压标准差
            np.max(currents),  # 电流峰值
            np.polyfit(range(len(powers)), powers, 1)[0],  # 功率趋势
            # 更多特征...
        ])
