模块24：分布式锁与数据一致性

学习时长: 4-6 天
难度: ⭐⭐⭐⭐⭐
前置知识: 并发编程、Redis、分布式系统

一、问题背景

1.1 分布式场景下的数据一致性问题

单机环境:
  - synchronized / ReentrantLock
  - 本地内存可见
  - ACID事务保证

分布式环境:
  ❌ 多实例竞争
  ❌ 网络延迟
  ❌ 时钟不同步
  ❌ 脑裂问题
  ❌ 单点故障

解决方案:
  ✅ 分布式锁（Redis / ZooKeeper）
  ✅ 乐观锁（版本号）
  ✅ 最终一致性（消息队列）
  ✅ 分布式事务（Seata）

二、核心实现

2.1 DistributedLockService 完整实现

文件位置: backend/src/main/java/com/dormpower/cache/lock/DistributedLockService.java

package com.dormpower.cache.lock;

import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.data.redis.core.script.DefaultRedisScript;
import org.springframework.stereotype.Service;

import java.time.Duration;
import java.util.Collections;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.TimeUnit;

/**
 * Redis分布式锁（完整版）
 * 
 * 核心特性:
 * 1. 互斥性：SET NX保证
 * 2. 防死锁：过期时间保证
 * 3. 可重入：ThreadLocal存储锁信息
 * 4. WatchDog：自动续期机制
 * 5. 安全释放：Lua脚本原子操作
 * 
 * @author dormpower team
 * @version 2.0
 */
@Slf4j
@Service
public class DistributedLockService {

    private final RedisTemplate<String, Object> redisTemplate;

    // 可重入锁计数器
    private final Map<String, LockInfo> lockMap = new ConcurrentHashMap<>();

    // WatchDog线程池
    private final ScheduledExecutorService watchDogExecutor = 
        Executors.newScheduledThreadPool(10);

    // Lua脚本：原子释放锁
    private static final String UNLOCK_SCRIPT = 
        "if redis.call('get', KEYS[1]) == ARGV[1] then " +
        "    return redis.call('del', KEYS[1]) " +
        "else " +
        "    return 0 " +
        "end";

    // Lua脚本：续期锁
    private static final String RENEW_SCRIPT = 
        "if redis.call('get', KEYS[1]) == ARGV[1] then " +
        "    return redis.call('expire', KEYS[1], ARGV[2]) " +
        "else " +
        "    return 0 " +
        "end";

    public DistributedLockService(RedisTemplate<String, Object> redisTemplate) {
        this.redisTemplate = redisTemplate;
    }

    /**
     * 获取锁
     * 
     * @param lockKey 锁Key
     * @param waitTime 等待时间
     * @param leaseTime 锁持有时间
     * @return 锁标识
     */
    public String lock(String lockKey, Duration waitTime, Duration leaseTime) {
        // 检查可重入
        LockInfo lockInfo = lockMap.get(lockKey);
        if (lockInfo != null && lockInfo.isCurrentThread()) {
            lockInfo.increment();
            log.debug("Lock reentrant: {}", lockKey);
            return lockInfo.getLockValue();
        }

        String lockValue = UUID.randomUUID().toString();
        long endTime = System.currentTimeMillis() + waitTime.toMillis();

        while (System.currentTimeMillis() < endTime) {
            // SET NX EX 原子操作
            Boolean success = redisTemplate.opsForValue()
                .setIfAbsent(lockKey, lockValue, leaseTime);

            if (Boolean.TRUE.equals(success)) {
                log.info("Lock acquired: {} with value: {}", lockKey, lockValue);

                // 记录锁信息
                lockMap.put(lockKey, new LockInfo(lockValue));

                // 启动WatchDog
                startWatchDog(lockKey, lockValue, leaseTime);

                return lockValue;
            }

            // 等待100ms后重试
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                return null;
            }
        }

        log.warn("Lock timeout: {}", lockKey);
        return null;
    }

    /**
     * 释放锁
     */
    public boolean unlock(String lockKey, String lockValue) {
        // 检查可重入
        LockInfo lockInfo = lockMap.get(lockKey);
        if (lockInfo != null) {
            if (!lockInfo.getLockValue().equals(lockValue)) {
                throw new IllegalMonitorStateException("Cannot unlock with wrong lock value");
            }

            if (lockInfo.decrement() > 0) {
                log.debug("Lock reentrant decrement: {}", lockKey);
                return true;
            }

            // 计数器归零，真正释放
            lockMap.remove(lockKey);
        }

        // Lua脚本原子释放
        DefaultRedisScript<Long> script = new DefaultRedisScript<>(UNLOCK_SCRIPT, Long.class);
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(lockKey),
            lockValue
        );

        boolean success = result != null && result == 1;

        if (success) {
            log.info("Lock released: {}", lockKey);
        }

        return success;
    }

    /**
     * WatchDog自动续期
     */
    private void startWatchDog(String lockKey, String lockValue, Duration leaseTime) {
        // 每leaseTime/3时间续期一次
        long renewInterval = leaseTime.toMillis() / 3;

        watchDogExecutor.scheduleAtFixedRate(() -> {
            try {
                // 检查锁是否还存在
                LockInfo lockInfo = lockMap.get(lockKey);
                if (lockInfo == null || !lockInfo.getLockValue().equals(lockValue)) {
                    return;  // 锁已释放，停止续期
                }

                // 续期
                DefaultRedisScript<Long> script = new DefaultRedisScript<>(RENEW_SCRIPT, Long.class);
                redisTemplate.execute(
                    script,
                    Collections.singletonList(lockKey),
                    lockValue,
                    leaseTime.getSeconds()
                );

                log.debug("Lock renewed: {}", lockKey);
            } catch (Exception e) {
                log.error("Failed to renew lock: {}", lockKey, e);
            }
        }, renewInterval, renewInterval, TimeUnit.MILLISECONDS);
    }

    /**
     * 执行带锁的操作
     */
    public <T> T executeWithLock(String lockKey, Duration waitTime, Duration leaseTime, LockSupplier<T> supplier) {
        String lockValue = lock(lockKey, waitTime, leaseTime);

        if (lockValue == null) {
            throw new RuntimeException("Failed to acquire lock: " + lockKey);
        }

        try {
            return supplier.get();
        } finally {
            unlock(lockKey, lockValue);
        }
    }

    /**
     * 锁信息
     */
    private static class LockInfo {
        private final String lockValue;
        private final Thread ownerThread;
        private int count;

        public LockInfo(String lockValue) {
            this.lockValue = lockValue;
            this.ownerThread = Thread.currentThread();
            this.count = 1;
        }

        public boolean isCurrentThread() {
            return ownerThread == Thread.currentThread();
        }

        public String getLockValue() {
            return lockValue;
        }

        public int increment() {
            return ++count;
        }

        public int decrement() {
            return --count;
        }
    }

    @FunctionalInterface
    public interface LockSupplier<T> {
        T get() throws Exception;
    }
}

2.2 CacheInvalidationBroadcaster 缓存失效广播

文件位置: backend/src/main/java/com/dormpower/cache/consistency/CacheInvalidationBroadcaster.java

package com.dormpower.cache.consistency;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;

/**
 * 缓存失效广播服务
 * 
 * 问题:
 * 多实例部署时，一个实例更新数据库后，
 * 其他实例的本地缓存如何失效？
 * 
 * 解决方案:
 * Redis发布订阅机制
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class CacheInvalidationBroadcaster {

    private final RedisTemplate<String, Object> redisTemplate;

    private static final String CHANNEL = "cache:invalidation";

    /**
     * 发布缓存失效消息
     */
    public void publishInvalidation(String cacheKey) {
        redisTemplate.convertAndSend(CHANNEL, cacheKey);
        log.debug("Cache invalidation published: {}", cacheKey);
    }

    /**
     * 批量发布缓存失效
     */
    public void publishBatchInvalidation(String pattern) {
        redisTemplate.convertAndSend(CHANNEL, "pattern:" + pattern);
        log.debug("Batch cache invalidation published: {}", pattern);
    }
}

三、应用场景

3.1 设备控制命令防重

@Service
@RequiredArgsConstructor
public class CommandService {

    private final DistributedLockService distributedLock;

    public CommandResult executeCommand(String deviceId, Command cmd) {
        String lockKey = "command:lock:" + deviceId;

        // 使用分布式锁防止并发控制
        return distributedLock.executeWithLock(
            lockKey,
            Duration.ofSeconds(3),
            Duration.ofSeconds(10),
            () -> {
                // 检查设备状态
                Device device = deviceRepository.findById(deviceId);

                // 执行命令
                return doExecute(device, cmd);
            }
        );
    }
}

四、对比方案

分布式锁方案对比:
  Redis:
    优点: 高性能、易实现
    缺点: 单点故障、脑裂风险
    适用: 一般场景

  ZooKeeper:
    优点: 强一致性、高可用
    缺点: 性能较低、依赖ZK
    适用: 强一致性场景

  etcd:
    优点: 强一致性、现代架构
    缺点: 额外依赖
    适用: K8s生态

项目选择: Redis（性能优先）
