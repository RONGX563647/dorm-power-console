一、数据一致性挑战

1.1 问题场景

场景 1：设备状态与遥测数据不一致

设备上报状态: ONLINE
遥测数据显示: 离线数据（延迟）

场景 2：计费计算不准确

设备在账单周期内离线
但仍有遥测数据
导致计费错误

场景 3：并发修改冲突

两个管理员同时修改设备配置
后提交的覆盖了先提交的
导致配置丢失

二、解决方案

2.1 最终一致性设计

设备状态机:

public enum DeviceStatus {
    ONLINE,      // 在线
    OFFLINE,     // 离线
    FAULT,       // 故障
    PENDING,     // 待确认
    DECOMMISSIONED // 退役
}

@Service
public class DeviceStateManager {

    private static final Map<DeviceStatus, Set<DeviceStatus>> ALLOWED_TRANSITIONS = Map.of(
        DeviceStatus.ONLINE, Set.of(DeviceStatus.OFFLINE, DeviceStatus.FAULT, DeviceStatus.PENDING),
        DeviceStatus.OFFLINE, Set.of(DeviceStatus.ONLINE, DeviceStatus.DECOMMISSIONED),
        DeviceStatus.FAULT, Set.of(DeviceStatus.ONLINE, DeviceStatus.DECOMMISSIONED),
        DeviceStatus.PENDING, Set.of(DeviceStatus.ONLINE, DeviceStatus.OFFLINE),
        DeviceStatus.DECOMMISSIONED, Set.of() // 终态
    );

    @Transactional
    public void transitionStatus(String deviceId, DeviceStatus from, DeviceStatus to) {
        Device device = deviceRepository.findById(deviceId)
            .orElseThrow(() -> new DeviceNotFoundException(deviceId));

        if (!device.getStatus().equals(from)) {
            throw new InvalidStatusException(
                "设备状态不匹配: " + from + " -> " + to);
        }

        Set<DeviceStatus> allowed = ALLOWED_TRANSITIONS.get(from);
        if (!allowed.contains(to)) {
            throw new InvalidTransitionException(
                "不允许的状态转换: " + from + " -> " + to);
        }

        device.setStatus(to);
        device.setLastStatusChangeAt(LocalDateTime.now());
        deviceRepository.save(device);

        // 发布状态变更事件
        eventPublisher.publishEvent(new DeviceStatusChangedEvent(device));
    }
}

@Service
public class DeviceService {

    @Transactional
    public Device updateDevice(String deviceId, DeviceDTO dto) {
        Optional<Device> deviceOpt = deviceRepository.findById(deviceId);
        if (deviceOpt.isEmpty()) {
            throw new DeviceNotFoundException(deviceId);
        }

        Device device = deviceOpt.get();

        // 更新字段
        device.setName(dto.getName());
        device.setDescription(dto.getDescription());

        try {
            // 保存时会检查 version
            return deviceRepository.save(device);
        } catch (OptimisticLockingFailureException e) {
            throw new ConcurrentModificationException(
                "设备已被其他用户修改，请刷新后重试");
        }
    }
}

2.3 分布式事务

Saga 模式实现:

@Service
public class DeviceRegistrationSaga {

    private final DeviceService deviceService;
    private final RoomService roomService;
    private final MqttService mqttService;
    private final EventPublisher eventPublisher;

    @Transactional
    public void registerDevice(DeviceRegistrationRequest request) {
        Device device = null;
        try {
            // 1. 创建设备
            device = deviceService.createDevice(request);

            // 2. 绑定房间
            roomService.assignDeviceToRoom(device.getId(), request.getRoomId());

            // 3. 注册 MQTT 设备
            mqttService.registerDevice(device.getDeviceId());

            // 4. 发送注册成功事件
            eventPublisher.publishEvent(new DeviceRegisteredEvent(device));

        } catch (Exception e) {
            // 补偿操作
            if (device != null) {
                try {
                    deviceService.deleteDevice(device.getId());
                } catch (Exception ex) {
                    log.error("补偿操作失败", ex);
                }
            }
            throw new DeviceRegistrationException("设备注册失败", e);
        }
    }
}

三、数据校验

3.1 遥测数据校验

@Component
public class TelemetryDataValidator {

    private static final BigDecimal MAX_CURRENT = new BigDecimal("32.00");
    private static final BigDecimal MAX_VOLTAGE = new BigDecimal("250.00");
    private static final BigDecimal MAX_POWER = new BigDecimal("8000.00");

    public void validate(TelemetryData data) {
        if (data.getCurrent().compareTo(MAX_CURRENT) > 0) {
            throw new InvalidDataException("电流超过最大限制: " + MAX_CURRENT);
        }
        if (data.getVoltage().compareTo(MAX_VOLTAGE) > 0) {
            throw new InvalidDataException("电压超过最大限制: " + MAX_VOLTAGE);
        }
        if (data.getPower().compareTo(MAX_POWER) > 0) {
            throw new InvalidDataException("功率超过最大限制: " + MAX_POWER);
        }

        // 检查数据完整性
        if (data.getCurrent() == null || data.getVoltage() == null || data.getPower() == null) {
            throw new InvalidDataException("遥测数据不完整");
        }

        // 检查时间戳
        if (data.getTimestamp().isAfter(LocalDateTime.now().plusMinutes(5))) {
            throw new InvalidDataException("遥测数据时间戳异常");
        }
    }
}

3.2 计费数据校验

@Component
public class BillingDataValidator {

    private static final BigDecimal MAX_BILL_AMOUNT = new BigDecimal("10000.00");

    public void validate(Bill bill) {
        // 检查金额范围
        if (bill.getAmount().compareTo(MAX_BILL_AMOUNT) > 0) {
            throw new InvalidDataException("账单金额异常: " + bill.getAmount());
        }

        // 检查日期范围
        if (bill.getPeriodStart().isAfter(bill.getPeriodEnd())) {
            throw new InvalidDataException("账单周期无效");
        }

        // 检查关联数据
        if (bill.getDeviceId() == null && bill.getRoomId() == null) {
            throw new InvalidDataException("账单必须关联设备或房间");
        }
    }
}

4.2 手动修复工具

@RestController
@RequestMapping("/api/admin/repair")
@PreAuthorize("hasRole('ADMIN')")
public class DataRepairController {

    private final DataRepairService repairService;

    @PostMapping("/device-status")
    public ResponseEntity<?> repairDeviceStatus() {
        try {
            int repairedCount = repairService.repairDeviceStatus();
            return ResponseEntity.ok(Map.of("repaired", repairedCount));
        } catch (Exception e) {
            return ResponseEntity.status(500)
                .body(Map.of("error", e.getMessage()));
        }
    }

    @PostMapping("/telemetry/{deviceId}")
    public ResponseEntity<?> repairTelemetryData(@PathVariable String deviceId) {
        try {
            List<TelemetryData> repaired = repairService.repairTelemetryForDevice(deviceId);
            return ResponseEntity.ok(Map.of("repaired", repaired.size()));
        } catch (Exception e) {
            return ResponseEntity.status(500)
                .body(Map.of("error", e.getMessage()));
        }
    }
}
