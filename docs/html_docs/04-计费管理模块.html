二、需求分析与设计

2.1 业务场景

场景 1：月度账单生成

每月初定时任务 → 获取上月用电数据 → 应用电价规则 → 计算电费 → 生成账单 → 推送通知

场景 2：在线充值缴费

用户选择充值金额 → 调用支付接口 → 更新房间余额 → 自动扣缴欠费 → 记录充值流水

场景 3：电费试算

用户选择时间范围 → 获取用电数据 → 应用电价规则 → 试算电费 → 返回明细

场景 4：自动充值

定时检查余额 → 余额低于阈值 → 自动触发充值 → 更新余额 → 发送通知

2.3 技术选型与理由

技术 说明 版本 选型理由
BigDecimal 精确金额计算 - ✅ 精确计算：避免浮点数精度丢失问题
✅ 多种舍入模式：支持 8 种舍入模式（HALF_UP 等）
✅ 金融标准：金融行业强制使用
⚠️ 注意：实际项目使用 double，建议改为 BigDecimal
✅ 最佳实践：金额字段必须使用 BigDecimal，数据库用 DECIMAL 类型
Spring Scheduler 定时任务 3.2.x ✅ 月度账单：每月初自动生成上月账单
✅ 余额检查：定期检查房间余额，触发预警
✅ 简单配置：@Scheduled(cron = "0 0 0 1 * ?")
Strategy Pattern 电价策略模式 - ✅ 解耦算法：电价算法与业务逻辑分离
✅ 易扩展：新增电价策略无需修改现有代码
✅ 符合开闭原则：对扩展开放，对修改关闭
iText PDF 生成 7.x ✅ 账单导出：生成 PDF 格式账单
✅ 模板支持：支持 HTML/CSS 模板
⚠️ 替代方案：实际项目使用 EasyExcel 导出 Excel
EasyExcel Excel 导出 3.x ✅ 性能优秀：流式处理，避免 OOM
✅ 阿里开源：经过大规模生产验证
✅ 注解驱动：@ExcelProperty 简化配置

技术对比与取舍： - BigDecimal vs double：教学演示用 double 简化代码，实际项目必须用 BigDecimal - 定时任务 vs 消息队列：简单场景用定时任务，复杂场景用消息队列延迟处理 - PDF vs Excel：PDF 适合打印，Excel 适合数据分析，根据场景选择

package com.dormpower.model;

import jakarta.persistence.*;
import lombok.Data;
import java.math.BigDecimal;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.UUID;

@Data
@Entity
@Table(name = "electricity_bill", indexes = {
    @Index(name = "idx_bill_room_period", columnList = "roomId, billingPeriod"),
    @Index(name = "idx_bill_status", columnList = "status")
})
public class ElectricityBill {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(unique = true, nullable = false, length = 50)
    private String billNumber; // 账单编号

    @Column(name = "room_id", nullable = false)
    private UUID roomId;

    @Column(nullable = false, length = 7)
    private String billingPeriod; // 计费周期，如 2024-01

    @Column(nullable = false, precision = 10, scale = 2)
    private BigDecimal electricityAmount; // 用电量（kWh）

    @Column(nullable = false, precision = 10, scale = 4)
    private BigDecimal unitPrice; // 单价

    @Column(nullable = false, precision = 10, scale = 2)
    private BigDecimal totalAmount; // 总金额

    @Column(precision = 10, scale = 2)
    private BigDecimal paidAmount; // 已付金额

    @Column(precision = 10, scale = 2)
    private BigDecimal remainingAmount; // 剩余金额

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private BillStatus status = BillStatus.UNPAID;

    private LocalDate dueDate; // 截止日期

    private LocalDateTime paidAt; // 支付时间

    @Enumerated(EnumType.STRING)
    private PaymentMethod paymentMethod; // 支付方式

    private UUID paidBy; // 支付人

    private LocalDateTime createdAt = LocalDateTime.now();
    private LocalDateTime updatedAt = LocalDateTime.now();

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }
}

enum BillStatus {
    UNPAID, PAID, OVERDUE, CANCELLED
}

enum PaymentMethod {
    BALANCE, ALIPAY, WECHAT, BANK_TRANSFER
}

package com.dormpower.service.impl;

import com.dormpower.dto.BillingRequest;

import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.UUID;

@Service
@RequiredArgsConstructor
public class BillingServiceImpl implements BillingService {

    private final ElectricityBillRepository billRepository;
    private final TelemetryService telemetryService;

    private static final DateTimeFormatter FORMATTER = DateTimeFormatter.ofPattern("yyyy-MM");

    @Override
    @Transactional
    public ElectricityBill generateBill(BillingRequest request) {
        // 检查是否已存在账单
        String period = request.getBillingPeriod();
        UUID roomId = request.getRoomId();

        if (!request.isForceRegenerate()) {
            if (billRepository.findByRoomIdAndBillingPeriod(roomId, period).isPresent()) {
                throw new RuntimeException("账单已存在");
            }
        }

        // 计算电费
        LocalDateTime startTime = LocalDate.parse(period + "-01", DateTimeFormatter.ofPattern("yyyy-MM-dd"))
            .atStartOfDay();
        LocalDateTime endTime = startTime.plusMonths(1).minusSeconds(1);

        BillingResult result = calculateElectricity(roomId, startTime, endTime);

        // 生成账单
        ElectricityBill bill = new ElectricityBill();
        bill.setBillNumber("BILL-" + period + "-" + UUID.randomUUID().toString().substring(0, 8).toUpperCase());
        bill.setRoomId(roomId);
        bill.setBillingPeriod(period);
        bill.setElectricityAmount(result.getElectricityAmount());
        bill.setUnitPrice(result.getUnitPrice());
        bill.setTotalAmount(result.getTotalAmount());
        bill.setRemainingAmount(result.getTotalAmount());
        bill.setStatus(BillStatus.UNPAID);
        bill.setDueDate(LocalDate.now().plusDays(15));

        return billRepository.save(bill);
    }

    @Override
    public Page<ElectricityBill> getBills(UUID roomId, String period, Pageable pageable) {
        return billRepository.findByRoomId(roomId, pageable);
    }

    @Override
    @Transactional
    public ElectricityBill payBill(UUID billId, UUID userId, String paymentMethod) {
        ElectricityBill bill = billRepository.findById(billId)
            .orElseThrow(() -> new RuntimeException("账单不存在"));

        if (bill.getStatus() == BillStatus.PAID) {
            throw new RuntimeException("账单已支付");
        }

        // 更新账单状态
        bill.setStatus(BillStatus.PAID);
        bill.setPaidAt(LocalDateTime.now());
        bill.setPaidBy(userId);
        bill.setPaymentMethod(Enum.valueOf(PaymentMethod.class, paymentMethod));
        bill.setPaidAmount(bill.getRemainingAmount());
        bill.setRemainingAmount(BigDecimal.ZERO);

        return billRepository.save(bill);
    }

package com.dormpower.controller;

import com.dormpower.dto.BillingRequest;
import com.dormpower.dto.BillingResult;
import com.dormpower.model.ElectricityBill;
import com.dormpower.service.BillingService;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

import java.time.LocalDateTime;
import java.util.UUID;

@RestController
@RequestMapping("/api/bills")
@RequiredArgsConstructor
public class BillingController {

    private final BillingService billingService;

    @PostMapping("/generate")
    @PreAuthorize("hasAnyAuthority('bill:write', 'admin')")
    public ResponseEntity<ElectricityBill> generateBill(@RequestBody BillingRequest request) {
        return ResponseEntity.ok(billingService.generateBill(request));
    }

    @GetMapping
    @PreAuthorize("hasAnyAuthority('bill:read', 'admin')")
    public ResponseEntity<Page<ElectricityBill>> getBills(
        @RequestParam UUID roomId,
        @RequestParam(required = false) String period,
        Pageable pageable
    ) {
        return ResponseEntity.ok(billingService.getBills(roomId, period, pageable));
    }

    @PostMapping("/{id}/pay")
    @PreAuthorize("hasAnyAuthority('bill:write', 'admin')")
    public ResponseEntity<ElectricityBill> payBill(
        @PathVariable UUID id,
        @RequestParam UUID userId,
        @RequestParam String paymentMethod
    ) {
        return ResponseEntity.ok(billingService.payBill(id, userId, paymentMethod));
    }

    @GetMapping("/calculate")
    @PreAuthorize("hasAnyAuthority('bill:read', 'admin')")
    public ResponseEntity<BillingResult> calculateElectricity(
        @RequestParam UUID roomId,
        @RequestParam @DateTimeFormat(iso = DateTimeFormat.ISO.DATE_TIME) LocalDateTime startTime,
        @RequestParam @DateTimeFormat(iso = DateTimeFormat.ISO.DATE_TIME) LocalDateTime endTime
    ) {
        return ResponseEntity.ok(billingService.calculateElectricity(roomId, startTime, endTime));
    }
}

import axios from 'axios';
import type { ElectricityBill, BillingRequest, BillingResult, RoomBalance, RechargeRecord } from '@/types/billing';

const API_BASE = '/api/bills';

export const billingApi = {
  // 生成账单
  async generateBill(request: BillingRequest): Promise<ElectricityBill> {
    const response = await axios.post(API_BASE + '/generate', request);
    return response.data;
  },

  // 获取账单列表
  async getBills(roomId: string, period?: string, page = 0, size = 20) {
    const response = await axios.get(API_BASE, {
      params: { roomId, period, page, size }
    });
    return response.data;
  },

  // 支付账单
  async payBill(billId: string, userId: string, paymentMethod: string): Promise<ElectricityBill> {
    const response = await axios.post(`${API_BASE}/${billId}/pay`, null, {
      params: { userId, paymentMethod }
    });
    return response.data;
  },

  // 试算电费
  async calculateElectricity(
    roomId: string,
    startTime: string,
    endTime: string
  ): Promise<BillingResult> {
    const response = await axios.get(`${API_BASE}/calculate`, {
      params: { roomId, startTime, endTime }
    });
    return response.data;
  }
};

// 余额管理 API
const BALANCE_API_BASE = '/api/rooms';

export const balanceApi = {
  // 获取房间余额
  async getBalance(roomId: string): Promise<RoomBalance> {
    const response = await axios.get(`${BALANCE_API_BASE}/${roomId}/balance`);
    return response.data;
  },

  // 充值
  async recharge(roomId: string, amount: number, paymentMethod: string, remark?: string) {
    const response = await axios.post(`${BALANCE_API_BASE}/${roomId}/recharge`, {
      amount,
      paymentMethod,
      remark
    });
    return response.data;
  },

  // 获取充值记录
  async getRechargeRecords(roomId: string, page = 0, size = 20) {
    const response = await axios.get(`${BALANCE_API_BASE}/${roomId}/recharge-records`, {
      params: { page, size }
    });
    return response.data;
  }
};

<script setup lang="ts">
import { ref, reactive, onMounted } from 'vue';
import { billingApi } from '@/api/billing';
import { roomApi } from '@/api/room';
import type { ElectricityBill } from '@/types/billing';
import { ElMessage, ElMessageBox } from 'element-plus';

const rooms = ref<any[]>([]);
const bills = ref<ElectricityBill[]>([]);
const loading = ref(false);

const queryForm = reactive({
  roomId: '',
  status: ''
});

const currentPage = ref(1);
const pageSize = ref(20);
const total = ref(0);

// 加载房间列表
const loadRooms = async () => {
  try {
    const response = await roomApi.getRooms();
    rooms.value = response.content || response;
    if (rooms.value.length > 0) {
      queryForm.roomId = rooms.value[0].id;
    }
  } catch (error) {
    console.error('Failed to load rooms:', error);
  }
};

// 加载账单列表
const loadBills = async () => {
  if (!queryForm.roomId) return;

  loading.value = true;
  try {
    const response = await billingApi.getBills(
      queryForm.roomId,
      undefined,
      currentPage.value - 1,
      pageSize.value
    );

    bills.value = response.content || [];
    total.value = response.totalElements || 0;
  } catch (error: any) {
    ElMessage.error('加载账单失败：' + (error.response?.data?.message || ''));
  } finally {
    loading.value = false;
  }
};

// 查询
const handleQuery = () => {
  currentPage.value = 1;
  loadBills();
};

// 生成账单
const handleGenerate = async () => {
  try {
    await ElMessageBox.prompt('请输入计费周期（如 2024-01）', '生成账单', {
      inputPattern: /^\d{4}-\d{2}$/,
      inputErrorMessage: '格式不正确'
    });

    await billingApi.generateBill({
      roomId: queryForm.roomId,
      billingPeriod: '2024-01',
      forceRegenerate: false
    });

    ElMessage.success('账单生成成功');
    loadBills();
  } catch (error: any) {
    if (error !== 'cancel') {
      ElMessage.error('生成账单失败：' + (error.response?.data?.message || ''));
    }
  }
};

// 支付
const handlePay = async (bill: ElectricityBill) => {
  try {
    await ElMessageBox.confirm('确认支付该账单？', '支付确认');

    await billingApi.payBill(bill.id, 'current-user-id', 'BALANCE');

    ElMessage.success('支付成功');
    loadBills();
  } catch (error: any) {
    if (error !== 'cancel') {
      ElMessage.error('支付失败：' + (error.response?.data?.message || ''));
    }
  }
};

const handleViewDetail = (bill: ElectricityBill) => {
  ElMessage.info('账单详情功能开发中...');
};

const getStatusType = (status: string) => {
  const types: Record<string, any> = {
    UNPAID: 'warning',
    PAID: 'success',
    OVERDUE: 'danger'
  };
  return types[status] || 'info';
};

const getStatusText = (status: string) => {
  const texts: Record<string, string> = {
    UNPAID: '未支付',
    PAID: '已支付',
    OVERDUE: '逾期'
  };
  return texts[status] || status;
};

onMounted(() => {
  loadRooms();
  loadBills();
});
</script>
