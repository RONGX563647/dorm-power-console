2.1 AnomalyDetectionAgent 异常检测

文件位置: backend/src/main/java/com/dormpower/ml/AnomalyDetectionAgent.java

package com.dormpower.ml;

import com.dormpower.model.Telemetry;
import com.dormpower.service.AlertService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 异常检测Agent
 * 
 * 检测算法:
 * 1. 统计方法: 3-Sigma原则
 * 2. 规则引擎: 阈值判断
 * 3. 时序分析: 趋势异常
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class AnomalyDetectionAgent {

    private final AlertService alertService;

    // 设备历史数据缓存
    private final Map<String, List<Double>> deviceHistory = new ConcurrentHashMap<>();

    // 统计参数
    private final Map<String, Statistics> deviceStats = new ConcurrentHashMap<>();

    /**
     * 检测异常
     */
    public AnomalyResult detect(Telemetry telemetry) {
        String deviceId = telemetry.getDeviceId();

        // 1. 更新历史数据
        updateHistory(deviceId, telemetry.getPowerW());

        // 2. 更新统计参数
        Statistics stats = updateStatistics(deviceId);

        // 3. 多维度检测
        boolean isPowerAnomaly = detectPowerAnomaly(telemetry.getPowerW(), stats);
        boolean isVoltageAnomaly = detectVoltageAnomaly(telemetry.getVoltageV());
        boolean isCurrentAnomaly = detectCurrentAnomaly(telemetry.getCurrentA());

        // 4. 综合判断
        if (isPowerAnomaly || isVoltageAnomaly || isCurrentAnomaly) {
            return AnomalyResult.builder()
                .isAnomaly(true)
                .powerAnomaly(isPowerAnomaly)
                .voltageAnomaly(isVoltageAnomaly)
                .currentAnomaly(isCurrentAnomaly)
                .build();
        }

        return AnomalyResult.normal();
    }

    /**
     * 功率异常检测（3-Sigma原则）
     */
    private boolean detectPowerAnomaly(double power, Statistics stats) {
        if (stats.count < 30) {
            return false;  // 数据不足
        }

        // 3-Sigma原则: 超出均值±3倍标准差为异常
        double upper = stats.mean + 3 * stats.stdDev;
        double lower = stats.mean - 3 * stats.stdDev;

        return power > upper || power < lower;
    }

    /**
     * 电压异常检测（阈值判断）
     */
    private boolean detectVoltageAnomaly(double voltage) {
        // 正常范围: 198V - 242V (220V ± 10%)
        return voltage < 198 || voltage > 242;
    }

    /**
     * 电流异常检测
     */
    private boolean detectCurrentAnomaly(double current) {
        // 过流保护: > 20A
        return current > 20;
    }

    /**
     * 更新历史数据
     */
    private void updateHistory(String deviceId, double value) {
        deviceHistory.computeIfAbsent(deviceId, k -> new java.util.ArrayList<>())
            .add(value);

        // 保留最近1000条数据
        List<Double> history = deviceHistory.get(deviceId);
        if (history.size() > 1000) {
            history.subList(0, history.size() - 1000).clear();
        }
    }

    /**
     * 更新统计参数
     */
    private Statistics updateStatistics(String deviceId) {
        List<Double> history = deviceHistory.get(deviceId);
        if (history == null || history.isEmpty()) {
            return new Statistics();
        }

        double mean = history.stream().mapToDouble(Double::doubleValue).average().orElse(0);
        double stdDev = Math.sqrt(
            history.stream()
                .mapToDouble(v -> Math.pow(v - mean, 2))
                .average()
                .orElse(0)
        );

        Statistics stats = new Statistics();
        stats.mean = mean;
        stats.stdDev = stdDev;
        stats.count = history.size();

        deviceStats.put(deviceId, stats);
        return stats;
    }

    @lombok.Data
    static class Statistics {
        double mean = 0;
        double stdDev = 0;
        int count = 0;
    }

    @lombok.Data
    @lombok.Builder
    public static class AnomalyResult {
        boolean isAnomaly;
        boolean powerAnomaly;
        boolean voltageAnomaly;
        boolean currentAnomaly;

        static AnomalyResult normal() {
            return AnomalyResult.builder()
                .isAnomaly(false)
                .build();
        }
    }
}

2.2 PowerPredictionAgent 功率预测

文件位置: backend/src/main/java/com/dormpower/ml/PowerPredictionAgent.java

package com.dormpower.ml;

import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;

/**
 * 功率预测Agent
 * 
 * 预测方法:
 * 1. 移动平均法（短期）
 * 2. 线性回归（中期）
 * 3. 季节性分解（长期）
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
public class PowerPredictionAgent {

    /**
     * 短期预测（1小时）- 移动平均
     */
    public double predictShortTerm(List<Double> history, int steps) {
        if (history.isEmpty()) {
            return 0;
        }

        // 简单移动平均
        int windowSize = Math.min(6, history.size());  // 取最近6个数据点
        List<Double> recent = history.subList(
            history.size() - windowSize, history.size()
        );

        double sum = recent.stream().mapToDouble(Double::doubleValue).sum();
        return sum / windowSize;
    }

    /**
     * 中期预测（24小时）- 线性回归
     */
    public double[] predictMidTerm(List<Double> history, int steps) {
        if (history.size() < 2) {
            return new double[steps];
        }

        // 简单线性回归: y = a + bx
        int n = history.size();
        double sumX = 0, sumY = 0, sumXY = 0, sumX2 = 0;

        for (int i = 0; i < n; i++) {
            sumX += i;
            sumY += history.get(i);
            sumXY += i * history.get(i);
            sumX2 += i * i;
        }

        double slope = (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
        double intercept = (sumY - slope * sumX) / n;

        // 预测未来steps个点
        double[] predictions = new double[steps];
        for (int i = 0; i < steps; i++) {
            predictions[i] = intercept + slope * (n + i);
        }

        return predictions;
    }
}

3.1 MLUtils 工具类

文件位置: backend/src/main/java/com/dormpower/ml/MLUtils.java

package com.dormpower.ml;

import java.util.List;

/**
 * 机器学习工具类
 * 
 * @author dormpower team
 * @version 1.0
 */
public class MLUtils {

    /**
     * 计算统计特征
     */
    public static FeatureSet extractFeatures(List<Double> data) {
        FeatureSet features = new FeatureSet();

        // 基础统计
        features.mean = data.stream().mapToDouble(Double::doubleValue).average().orElse(0);
        features.max = data.stream().mapToDouble(Double::doubleValue).max().orElse(0);
        features.min = data.stream().mapToDouble(Double::doubleValue).min().orElse(0);
        features.stdDev = calculateStdDev(data, features.mean);

        // 趋势特征
        features.trend = calculateTrend(data);

        // 波动特征
        features.volatility = calculateVolatility(data);

        return features;
    }

    private static double calculateStdDev(List<Double> data, double mean) {
        return Math.sqrt(
            data.stream()
                .mapToDouble(v -> Math.pow(v - mean, 2))
                .average()
                .orElse(0)
        );
    }

    private static double calculateTrend(List<Double> data) {
        if (data.size() < 2) return 0;
        double first = data.get(0);
        double last = data.get(data.size() - 1);
        return (last - first) / first;
    }

    private static double calculateVolatility(List<Double> data) {
        if (data.size() < 2) return 0;
        double sum = 0;
        for (int i = 1; i < data.size(); i++) {
            sum += Math.abs(data.get(i) - data.get(i-1));
        }
        return sum / (data.size() - 1);
    }

    public static class FeatureSet {
        public double mean;
        public double max;
        public double min;
        public double stdDev;
        public double trend;
        public double volatility;
    }
}
