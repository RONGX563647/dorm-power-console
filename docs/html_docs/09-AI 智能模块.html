二、需求分析与设计

2.1 业务场景

场景 1：AI 用电报告生成

用户请求报告 → 获取用电数据 → AI 分析 → 生成报告 → 返回用户

场景 2：智能问答

用户提问 → NLP 处理 → 意图识别 → 查询数据 → 生成回答

场景 3：用电异常分析

采集用电数据 → 异常检测算法 → 发现异常 → 生成分析报告

2.2 功能需求清单

功能 描述 优先级
AI 报告生成 月度/周度用电报告 P0
智能问答 用电咨询问答 P1
用电分析 用电行为分析 P1
用电预测 未来用电预测 P2
节能建议 智能节能建议 P2

2.3 技术选型与理由

技术 说明 版本 选型理由
Spring AI AI 服务集成框架 1.x ✅ 统一接口：抽象多家 AI 服务商接口
✅ 开箱即用：简化 AI 服务集成流程
✅ 流式响应：支持 SSE 流式输出
⚠️ 替代方案：教学场景可直接调用 REST API
OpenAI API 大模型服务 - ✅ 智能问答：GPT 模型提供自然语言问答
✅ 报告生成：根据用电数据生成分析报告
✅ 灵活配置：通过 Prompt 工程控制输出
⚠️ 替代方案：国内可用文心一言/通义千问
Smile Java 机器学习库 3.x ✅ 预测算法：支持回归、时间序列预测
✅ 轻量级：纯 Java 实现，无外部依赖
✅ 用电预测：基于历史数据预测未来用电
⚠️ 简化实现：教学场景可用简单线性回归
ONNX Runtime 模型推理引擎 1.x ✅ 跨平台：支持多种框架训练的模型
✅ 高性能：优化的推理引擎
✅ 异常检测：加载预训练模型检测用电异常
⚠️ 学习曲线：需要了解模型训练和部署流程

技术对比与取舍： - 调用 API vs 自建模型：教学场景调用 API 简化实现，生产环境考虑自建模型降低成本 - 复杂模型 vs 简单算法：根据数据量选择，小数据量用简单算法，大数据量用深度学习 - 实时预测 vs 离线分析：实时性要求高用流式处理，否则用离线批处理

package com.dormpower.model;

import jakarta.persistence.*;
import lombok.Data;
import java.time.LocalDateTime;
import java.util.UUID;

@Data
@Entity
@Table(name = "ai_report")
public class AIReport {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(name = "room_id", nullable = false)
    private UUID roomId;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private ReportType reportType = ReportType.MONTHLY;

    @Column(nullable = false, length = 20)
    private String reportPeriod; // 如 2024-01

    @Column(columnDefinition = "TEXT")
    private String reportContent; // JSON 格式报告内容

    @Column(length = 2000)
    private String summary; // 报告摘要

    @Column(precision = 10, scale = 2)
    private java.math.BigDecimal totalElectricity; // 总用电量

    @Column(precision = 10, scale = 2)
    private java.math.BigDecimal totalCost; // 总电费

    @Column(precision = 10, scale = 2)
    private java.math.BigDecimal predictedNext; // 预测下期

    private Integer energySavingScore; // 节能评分（0-100）

    @Column(length = 1000)
    private String suggestions; // 节能建议

    private LocalDateTime generatedAt = LocalDateTime.now();

    @Column(name = "generated_by")
    private UUID generatedBy;
}

enum ReportType {
    DAILY, WEEKLY, MONTHLY, YEARLY
}

步骤 4：Controller 层

4.1 创建 AIReportController

文件位置: backend/src/main/java/com/dormpower/controller/AIReportController.java

package com.dormpower.controller;

import com.dormpower.model.AIReport;
import com.dormpower.service.AIReportService;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

import java.util.List;
import java.util.UUID;

@RestController
@RequestMapping("/api/ai-reports")
@RequiredArgsConstructor
public class AIReportController {

    private final AIReportService aiReportService;

    @PostMapping("/generate")
    @PreAuthorize("hasAnyAuthority('ai:write', 'admin')")
    public ResponseEntity<AIReport> generateReport(
        @RequestParam UUID roomId,
        @RequestParam String period
    ) {
        return ResponseEntity.ok(aiReportService.generateReport(roomId, period));
    }

    @GetMapping
    @PreAuthorize("hasAnyAuthority('ai:read', 'admin')")
    public ResponseEntity<AIReport> getReport(
        @RequestParam UUID roomId,
        @RequestParam String period
    ) {
        return aiReportService.getReport(roomId, period)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }

    @GetMapping("/history")
    @PreAuthorize("hasAnyAuthority('ai:read', 'admin')")
    public ResponseEntity<List<AIReport>> getReportHistory(@RequestParam UUID roomId) {
        return ResponseEntity.ok(aiReportService.getReportHistory(roomId));
    }

    @GetMapping("/suggestion")
    @PreAuthorize("hasAnyAuthority('ai:read', 'admin')")
    public ResponseEntity<String> getSuggestion(@RequestParam UUID roomId) {
        return ResponseEntity.ok(aiReportService.getEnergySavingSuggestion(roomId));
    }
}

4.2 API 封装

文件位置: frontend/src/api/ai.ts

import axios from 'axios';
import type { AIReport, ChatMessage, ChatRequest } from '@/types/ai';

const REPORT_API_BASE = '/api/ai-reports';
const CHAT_API_BASE = '/api/ai/chat';

export const aiReportApi = {
  async generateReport(roomId: string, period: string): Promise<AIReport> {
    const response = await axios.post(`${REPORT_API_BASE}/generate`, null, {
      params: { roomId, period }
    });
    return response.data;
  },

  async getReport(roomId: string, period: string): Promise<AIReport> {
    const response = await axios.get(REPORT_API_BASE, {
      params: { roomId, period }
    });
    return response.data;
  },

  async getReportHistory(roomId: string): Promise<AIReport[]> {
    const response = await axios.get(`${REPORT_API_BASE}/history`, {
      params: { roomId }
    });
    return response.data;
  },

  async getSuggestion(roomId: string): Promise<string> {
    const response = await axios.get(`${REPORT_API_BASE}/suggestion`, {
      params: { roomId }
    });
    return response.data;
  }
};

export const aiChatApi = {
  async sendMessage(request: ChatRequest): Promise<ChatMessage> {
    const response = await axios.post(CHAT_API_BASE, request);
    return response.data;
  }
};

4.3 AI 报告页面

文件位置: frontend/src/views/AIReportView.vue

<template>
  <div class="ai-report">
    <el-card>
      <template #header>
        <span>AI 用电报告</span>
        <el-button type="primary" @click="handleGenerate">生成报告</el-button>
      </template>

      <div v-if="report" class="report-content">
        <el-descriptions title="报告概览" :column="2" border>
          <el-descriptions-item label="用电周期">{{ report.reportPeriod }}</el-descriptions-item>
          <el-descriptions-item label="总用电量">{{ report.totalElectricity }} kWh</el-descriptions-item>
          <el-descriptions-item label="总电费">{{ report.totalCost }} 元</el-descriptions-item>
          <el-descriptions-item label="节能评分">
            <el-tag :type="getScoreType(report.energySavingScore)">
              {{ report.energySavingScore }}分
            </el-tag>
          </el-descriptions-item>
        </el-descriptions>

        <el-divider />

        <h3>用电分析</h3>
        <p>{{ report.summary }}</p>

        <h3>节能建议</h3>
        <p>{{ report.suggestions }}</p>
      </div>

      <el-empty v-else description="暂无报告，请先生成报告" />
    </el-card>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { aiReportApi } from '@/api/ai';
import type { AIReport } from '@/types/ai';
import { ElMessage } from 'element-plus';

const report = ref<AIReport | null>(null);

const loadReport = async () => {
  try {
    // TODO: 传入实际的 roomId 和 period
    const result = await aiReportApi.getReport('room-id', '2024-01');
    report.value = result;
  } catch (error) {
    console.log('No report found');
  }
};

const handleGenerate = async () => {
  try {
    report.value = await aiReportApi.generateReport('room-id', '2024-01');
    ElMessage.success('报告生成成功');
  } catch (error: any) {
    ElMessage.error('生成失败：' + (error.response?.data?.message || ''));
  }
};

const getScoreType = (score?: number) => {
  if (!score) return 'info';
  if (score >= 90) return 'success';
  if (score >= 70) return 'warning';
  return 'danger';
};

onMounted(() => {
  loadReport();
});
</script>
