二、需求分析与设计

2.1 业务场景

场景 1：楼栋管理

创建楼栋 → 设置楼层数 → 设置房间数 → 楼栋信息维护

场景 2：房间管理

创建房间 → 设置床位 → 分配设备 → 房间信息维护

场景 3：入住管理

学生申请 → 管理员审核 → 分配房间 → 办理入住 → 设备绑定

场景 4：退宿管理

学生申请 → 管理员审核 → 检查设备 → 办理退宿 → 清理数据

2.2 功能需求清单

功能 描述 优先级
楼栋管理 CRUD 操作 P0
房间管理 CRUD 操作 P0
入住管理 入住办理 P0
退宿管理 退宿办理 P0
住宿统计 入住率统计 P1
设备绑定 房间设备关联 P2

2.3 技术选型与理由

技术 说明 版本 选型理由
JPA/Hibernate ORM 框架 3.2.x ✅ 级联操作：楼栋 - 房间 - 学生层级关系自动管理
✅ 查询简化：Spring Data JPA 方法命名查询
✅ 事务管理：@Transactional 保证数据一致性
Spring Data REST RESTful API 3.2.x ✅ 快速开发：Repository 自动暴露为 REST API
✅ HATEOAS：支持资源关联链接
⚠️ 简化实现：教学场景手动编写 Controller 更清晰
Vue Element Plus UI 组件库 2.x ✅ 表格组件：ElTable 支持分页、排序、筛选
✅ 表单组件：ElForm 支持验证、布局
✅ 对话框：ElDialog 实现弹窗编辑
EasyExcel Excel 导入导出 3.x ✅ 批量导入：支持 Excel 批量导入学生信息
✅ 模板导出：导出宿舍分配报表
✅ 流式处理：避免大数据量 OOM

技术对比与取舍： - 单表 vs 层级设计：选择楼栋 - 房间两层设计，符合实际业务场景 - 手动 SQL vs JPA：选择 JPA 因为 CRUD 操作多，查询逻辑简单 - 前端分页 vs 后端分页：选择后端分页，减少前端压力，适合大数据量

// Building.java
package com.dormpower.model;

import jakarta.persistence.*;
import lombok.Data;
import java.time.LocalDateTime;
import java.util.UUID;

@Data
@Entity
@Table(name = "building")
public class Building {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(nullable = false, length = 100)
    private String name;

    @Column(length = 500)
    private String address;

    private Integer floors; // 楼层数

    private Integer totalRooms; // 房间总数

    @Column(length = 500)
    private String description;

    @Column(nullable = false)
    private boolean enabled = true;

    private LocalDateTime createdAt = LocalDateTime.now();
    private LocalDateTime updatedAt = LocalDateTime.now();

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }
}

// Room.java
package com.dormpower.model;

import jakarta.persistence.*;
import lombok.Data;
import java.time.LocalDateTime;
import java.util.UUID;

@Data
@Entity
@Table(name = "room", indexes = {
    @Index(name = "idx_room_building", columnList = "building_id"),
    @Index(name = "idx_room_number", columnList = "room_number")
})
public class Room {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(name = "room_number", nullable = false, length = 50)
    private String roomNumber;

    @Column(name = "building_id", nullable = false)
    private UUID buildingId;

    private Integer floor; // 楼层

    private Integer maxOccupancy = 4; // 最大入住人数

    private Integer currentOccupancy = 0; // 当前入住人数

    @Enumerated(EnumType.STRING)
    private RoomStatus status = RoomStatus.VACANT;

    @Column(length = 500)
    private String description;

    private LocalDateTime createdAt = LocalDateTime.now();
    private LocalDateTime updatedAt = LocalDateTime.now();

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }
}

enum RoomStatus {
    VACANT,     // 空置
    OCCUPIED,   // 已入住
    MAINTENANCE // 维修中
}

package com.dormpower.model;

import jakarta.persistence.*;
import lombok.Data;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.UUID;

@Data
@Entity
@Table(name = "residence_record")
public class ResidenceRecord {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(name = "user_id", nullable = false)
    private UUID userId;

    @Column(name = "room_id", nullable = false)
    private UUID roomId;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private ResidenceStatus status = ResidenceStatus.ACTIVE;

    private LocalDate checkInDate; // 入住日期
    private LocalDate checkOutDate; // 退宿日期

    @Column(length = 500)
    private String remark;

    private LocalDateTime createdAt = LocalDateTime.now();
    private LocalDateTime updatedAt = LocalDateTime.now();

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }
}

enum ResidenceStatus {
    ACTIVE,     // 在住
    CHECKED_OUT // 已退宿
}

package com.dormpower.repository;

import com.dormpower.model.Room;
import com.dormpower.model.RoomStatus;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;

import java.util.List;
import java.util.Optional;
import java.util.UUID;

@Repository
public interface RoomRepository extends JpaRepository<Room, UUID> {

    List<Room> findByBuildingId(UUID buildingId);

    List<Room> findByStatus(RoomStatus status);

    @Query("SELECT COUNT(r) FROM Room r WHERE r.buildingId = :buildingId AND r.status = 'OCCUPIED'")
    Long countOccupiedRooms(@Param("buildingId") UUID buildingId);

    Optional<Room> findByRoomNumber(String roomNumber);
}

4.1 创建 RoomController

文件位置: backend/src/main/java/com/dormpower/controller/RoomController.java

package com.dormpower.controller;

import com.dormpower.dto.RoomDTO;
import com.dormpower.model.Room;
import com.dormpower.service.RoomService;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.http.ResponseEntity;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

import java.util.List;
import java.util.UUID;

@RestController
@RequestMapping("/api/rooms")
@RequiredArgsConstructor
public class RoomController {

    private final RoomService roomService;

    @PostMapping
    @PreAuthorize("hasAnyAuthority('room:write', 'admin')")
    public ResponseEntity<Room> createRoom(@RequestBody RoomDTO dto) {
        return ResponseEntity.ok(roomService.createRoom(dto));
    }

    @GetMapping
    @PreAuthorize("hasAnyAuthority('room:read', 'admin')")
    public ResponseEntity<Page<Room>> getRooms(
        @RequestParam(required = false) UUID buildingId,
        Pageable pageable
    ) {
        return ResponseEntity.ok(roomService.getRooms(buildingId, pageable));
    }

    @GetMapping("/{id}")
    @PreAuthorize("hasAnyAuthority('room:read', 'admin')")
    public ResponseEntity<Room> getRoom(@PathVariable UUID id) {
        return roomService.getRoom(id)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }

    @PutMapping("/{id}")
    @PreAuthorize("hasAnyAuthority('room:write', 'admin')")
    public ResponseEntity<Room> updateRoom(
        @PathVariable UUID id,
        @RequestBody RoomDTO dto
    ) {
        return ResponseEntity.ok(roomService.updateRoom(id, dto));
    }

    @DeleteMapping("/{id}")
    @PreAuthorize("hasAnyAuthority('room:write', 'admin')")
    public ResponseEntity<Void> deleteRoom(@PathVariable UUID id) {
        roomService.deleteRoom(id);
        return ResponseEntity.noContent().build();
    }

    @PostMapping("/{id}/checkin")
    @PreAuthorize("hasAnyAuthority('room:write', 'admin')")
    public ResponseEntity<Room> checkIn(
        @PathVariable UUID id,
        @RequestParam UUID userId
    ) {
        return ResponseEntity.ok(roomService.checkIn(id, userId));
    }

    @PostMapping("/{id}/checkout")
    @PreAuthorize("hasAnyAuthority('room:write', 'admin')")
    public ResponseEntity<Room> checkOut(
        @PathVariable UUID id,
        @RequestParam UUID userId
    ) {
        return ResponseEntity.ok(roomService.checkOut(id, userId));
    }
}

4.2 API 封装

文件位置: frontend/src/api/room.ts

import axios from 'axios';
import type { Room, Building, ResidenceRecord } from '@/types/room';

const ROOM_API_BASE = '/api/rooms';
const BUILDING_API_BASE = '/api/buildings';

export const roomApi = {
  async getRooms(buildingId?: string, page = 0, size = 20) {
    const response = await axios.get(ROOM_API_BASE, {
      params: { buildingId, page, size }
    });
    return response.data;
  },

  async createRoom(room: Partial<Room>) {
    const response = await axios.post(ROOM_API_BASE, room);
    return response.data;
  },

  async updateRoom(roomId: string, room: Partial<Room>) {
    const response = await axios.put(`${ROOM_API_BASE}/${roomId}`, room);
    return response.data;
  },

  async deleteRoom(roomId: string) {
    await axios.delete(`${ROOM_API_BASE}/${roomId}`);
  },

  async checkIn(roomId: string, userId: string) {
    const response = await axios.post(`${ROOM_API_BASE}/${roomId}/checkin`, null, {
      params: { userId }
    });
    return response.data;
  },

  async checkOut(roomId: string, userId: string) {
    const response = await axios.post(`${ROOM_API_BASE}/${roomId}/checkout`, null, {
      params: { userId }
    });
    return response.data;
  }
};

export const buildingApi = {
  async getBuildings() {
    const response = await axios.get(BUILDING_API_BASE);
    return response.data;
  },

  async createBuilding(building: Partial<Building>) {
    const response = await axios.post(BUILDING_API_BASE, building);
    return response.data;
  }
};
