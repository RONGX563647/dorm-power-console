模块16：多级缓存系统设计与实现

学习时长: 5-7 天
难度: ⭐⭐⭐⭐⭐
前置知识: Redis基础、Spring Cache、并发编程

一、系统架构概述

1.1 为什么需要多级缓存？

单一缓存的问题:
  ❌ Redis单点故障 → 所有缓存不可用
  ❌ 网络延迟 → 每次请求都要访问Redis
  ❌ 缓存穿透 → 恶意查询打穿到数据库
  ❌ 缓存雪崩 → 大量缓存同时过期
  ❌ 热点Key → 单个Key被高频访问

多级缓存解决方案:
  ✅ L1: Caffeine本地缓存（微秒级）
  ✅ L2: Redis分布式缓存（毫秒级）
  ✅ L3: 数据库（兜底）
  ✅ 布隆过滤器（防穿透）
  ✅ 分布式锁（防击穿）
  ✅ 随机过期时间（防雪崩）
  ✅ 热Key检测与分离（防热点）

2.1 MultiLevelCache 多级缓存核心

文件位置: backend/src/main/java/com/dormpower/cache/MultiLevelCache.java

package com.dormpower.cache;

import com.dormpower.cache.bloom.BloomFilterService;
import com.dormpower.cache.hotkey.HotKeyDetectorService;
import com.dormpower.cache.lock.DistributedLockService;
import com.dormpower.cache.protection.CacheProtectionService;
import com.github.benmanes.caffeine.cache.Cache;
import com.github.benmanes.caffeine.cache.Caffeine;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Component;

import java.time.Duration;
import java.util.concurrent.Callable;
import java.util.concurrent.TimeUnit;

/**
 * 多级缓存核心实现
 * 
 * 架构设计:
 * L1: Caffeine本地缓存（微秒级，容量有限）
 * L2: Redis分布式缓存（毫秒级，容量大）
 * L3: 数据库（兜底，最慢）
 * 
 * 防护机制:
 * 1. 布隆过滤器 → 防缓存穿透
 * 2. 分布式锁 → 防缓存击穿
 * 3. 随机过期时间 → 防缓存雪崩
 * 4. 热Key检测 → 防热点Key问题
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class MultiLevelCache {

    // L1: Caffeine本地缓存
    private final Cache<String, Object> localCache;

    // L2: Redis分布式缓存
    private final RedisTemplate<String, Object> redisTemplate;

    // 防护组件
    private final BloomFilterService bloomFilter;
    private final DistributedLockService distributedLock;
    private final HotKeyDetectorService hotKeyDetector;
    private final CacheProtectionService protectionService;

    /**
     * 获取缓存数据（多级缓存查询）
     * 
     * @param key 缓存键
     * @param loader 数据加载器（数据库查询）
     * @param ttl 过期时间
     * @return 缓存数据
     */
    @SuppressWarnings("unchecked")
    public <T> T get(String key, Callable<T> loader, Duration ttl) {
        // 1. 检查L1缓存
        T value = (T) localCache.getIfPresent(key);
        if (value != null) {
            log.debug("L1 Cache Hit: {}", key);
            hotKeyDetector.recordAccess(key);  // 记录访问
            return value;
        }

        // 2. 检查布隆过滤器（防穿透）
        if (!bloomFilter.mightContain(key)) {
            log.debug("Bloom Filter Miss: {}", key);
            return null;
        }

        // 3. 检查L2缓存
        value = (T) redisTemplate.opsForValue().get(key);
        if (value != null) {
            log.debug("L2 Cache Hit: {}", key);
            // 回填L1缓存
            localCache.put(key, value);
            hotKeyDetector.recordAccess(key);
            return value;
        }

        // 4. 缓存未命中，使用分布式锁防止击穿
        return protectionService.executeWithProtection(key, () -> {
            // 5. 再次检查L2缓存（双重检查）
            value = (T) redisTemplate.opsForValue().get(key);
            if (value != null) {
                localCache.put(key, value);
                return value;
            }

            // 6. 从数据库加载
            log.info("Cache Miss, Loading from DB: {}", key);
            value = loader.call();

            if (value != null) {
                // 7. 写入L2缓存（加随机过期时间防雪崩）
                Duration randomTtl = protectionService.addJitter(ttl);
                redisTemplate.opsForValue().set(key, value, randomTtl);

                // 8. 写入L1缓存
                localCache.put(key, value);

                // 9. 加入布隆过滤器
                bloomFilter.add(key);
            }

            hotKeyDetector.recordAccess(key);
            return value;
        });
    }

    /**
     * 删除缓存（双删策略）
     */
    public void delete(String key) {
        // 1. 删除L1缓存
        localCache.invalidate(key);

        // 2. 删除L2缓存
        redisTemplate.delete(key);

        // 3. 延迟再次删除（防止并发问题）
        CompletableFuture.runAsync(() -> {
            try {
                Thread.sleep(500);
                redisTemplate.delete(key);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        });
    }

    /**
     * 更新缓存
     */
    public void put(String key, Object value, Duration ttl) {
        // 1. 更新L2缓存
        Duration randomTtl = protectionService.addJitter(ttl);
        redisTemplate.opsForValue().set(key, value, randomTtl);

        // 2. 更新L1缓存
        localCache.put(key, value);

        // 3. 更新布隆过滤器
        bloomFilter.add(key);
    }
}

2.2 BloomFilter 布隆过滤器

文件位置: backend/src/main/java/com/dormpower/cache/bloom/RedisBloomFilter.java

package com.dormpower.cache.bloom;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Component;

/**
 * Redis布隆过滤器实现
 * 
 * 原理:
 * 1. 使用多个Hash函数将元素映射到BitSet
 * 2. 查询时检查所有Hash位是否都为1
 * 3. 如果全为1 → 可能存在（有误判率）
 * 4. 如果有0 → 一定不存在
 * 
 * 优点:
 * - 空间效率极高（1亿数据仅需11MB）
 * - 查询时间复杂度O(k)，k为Hash函数数量
 * - 不会漏判
 * 
 * 缺点:
 * - 有误判率（可配置）
 * - 不支持删除（需要Counting Bloom Filter）
 * 
 * 应用场景:
 * - 防止缓存穿透
 * - 恶意请求过滤
 * - 黑名单检查
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class RedisBloomFilter {

    private final RedisTemplate<String, Object> redisTemplate;

    // 布隆过滤器参数
    private static final int EXPECTED_INSERTIONS = 1000000;  // 预期插入100万
    private static final double FPP = 0.01;                   // 误判率1%

    /**
     * 判断元素是否可能存在
     * 
     * @param key 元素
     * @return true=可能存在，false=一定不存在
     */
    public boolean mightContain(String key) {
        String filterKey = "bloom:dormpower";

        // 使用Redis的BitSet实现
        // 这里简化为Hash方式
        int[] hashPositions = getHashPositions(key);

        for (int pos : hashPositions) {
            Boolean bit = redisTemplate.opsForValue()
                .getBit(filterKey, pos);
            if (bit == null || !bit) {
                return false;  // 有一位为0，一定不存在
            }
        }

        return true;  // 所有位都为1，可能存在
    }

    /**
     * 添加元素到布隆过滤器
     */
    public void add(String key) {
        String filterKey = "bloom:dormpower";

        int[] hashPositions = getHashPositions(key);

        for (int pos : hashPositions) {
            redisTemplate.opsForValue()
                .setBit(filterKey, pos, true);
        }

        log.debug("Added to BloomFilter: {}", key);
    }

    /**
     * 计算Hash位置
     * 
     * 使用双重Hash模拟多个Hash函数:
     * h_i(key) = h1(key) + i * h2(key)
     */
    private int[] getHashPositions(String key) {
        long hash1 = hash1(key);
        long hash2 = hash2(key);

        int numBits = getNumBits();
        int k = getNumHashFunctions();

        int[] positions = new int[k];
        for (int i = 0; i < k; i++) {
            positions[i] = (int) ((hash1 + i * hash2) % numBits);
        }

        return positions;
    }

    /**
     * Hash函数1 (MurmurHash3简化版)
     */
    private long hash1(String key) {
        long h = 0;
        for (char c : key.toCharArray()) {
            h = 31 * h + c;
        }
        return Math.abs(h);
    }

    /**
     * Hash函数2
     */
    private long hash2(String key) {
        long h = 0x9e3779b97f4a7c15L;  // 黄金比例
        for (char c : key.toCharArray()) {
            h ^= c;
            h *= 0x5bd1e995;
            h ^= h >>> 47;
        }
        return Math.abs(h);
    }

    /**
     * 计算BitSet位数
     * 
     * 公式: m = -n*ln(p)/(ln(2)^2)
     * n=预期插入数, p=误判率
     */
    private int getNumBits() {
        return (int) (-EXPECTED_INSERTIONS * Math.log(FPP) / (Math.log(2) * Math.log(2)));
    }

    /**
     * 计算Hash函数数量
     * 
     * 公式: k = m/n * ln(2)
     */
    private int getNumHashFunctions() {
        return (int) Math.ceil(getNumBits() / (double) EXPECTED_INSERTIONS * Math.log(2));
    }
}

2.3 HotKeyDetector 热Key检测

文件位置: backend/src/main/java/com/dormpower/cache/hotkey/HotKeyDetectorService.java

package com.dormpower.cache.hotkey;

import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.atomic.AtomicLong;

/**
 * 热Key检测服务
 * 
 * 问题背景:
 * 当某个Key被超高频访问时（如热点新闻），
 * 会导致该Key所在的Redis分片成为瓶颈。
 * 
 * 解决方案:
 * 1. 使用滑动窗口统计访问频率
 * 2. 超过阈值标记为热Key
 * 3. 热Key本地缓存，不经过Redis
 * 
 * 算法:
 * - 时间窗口: 1分钟
 * - 阈值: 1000次/分钟
 * - 热Key过期: 5分钟无访问自动降级
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
public class HotKeyDetectorService {

    // 访问计数器（线程安全）
    private final ConcurrentHashMap<String, AtomicLong> accessCounts = new ConcurrentHashMap<>();

    // 热Key集合
    private final ConcurrentHashMap<String, Long> hotKeys = new ConcurrentHashMap<>();

    // 配置参数
    private static final long HOT_KEY_THRESHOLD = 1000;  // 阈值：1000次/分钟
    private static final long WINDOW_SIZE_MS = 60000;    // 窗口：1分钟
    private static final long HOT_KEY_EXPIRE_MS = 300000; // 热Key过期：5分钟

    /**
     * 记录Key访问
     */
    public void recordAccess(String key) {
        // 1. 增加访问计数
        AtomicLong count = accessCounts.computeIfAbsent(key, k -> new AtomicLong(0));
        long currentCount = count.incrementAndGet();

        // 2. 检查是否达到热Key阈值
        if (currentCount >= HOT_KEY_THRESHOLD) {
            markAsHotKey(key);
        }

        // 3. 定期清理（每100次访问清理一次）
        if (currentCount % 100 == 0) {
            cleanup();
        }
    }

    /**
     * 标记为热Key
     */
    private void markAsHotKey(String key) {
        if (!hotKeys.containsKey(key)) {
            hotKeys.put(key, System.currentTimeMillis());
            log.warn("Hot Key Detected: {} (Access count exceeded threshold)", key);
        }
    }

    /**
     * 检查是否为热Key
     */
    public boolean isHotKey(String key) {
        Long addedTime = hotKeys.get(key);
        if (addedTime == null) {
            return false;
        }

        // 检查是否过期
        if (System.currentTimeMillis() - addedTime > HOT_KEY_EXPIRE_MS) {
            hotKeys.remove(key);
            return false;
        }

        return true;
    }

    /**
     * 清理过期数据
     */
    private void cleanup() {
        long now = System.currentTimeMillis();

        // 清理过期的热Key
        hotKeys.entrySet().removeIf(entry -> 
            now - entry.getValue() > HOT_KEY_EXPIRE_MS);

        // 清理访问计数器（简化版）
        if (accessCounts.size() > 10000) {
            accessCounts.clear();
        }
    }

    /**
     * 获取所有热Key
     */
    public Map<String, Long> getHotKeys() {
        cleanup();
        return Map.copyOf(hotKeys);
    }
}

2.4 DistributedLockService 分布式锁

文件位置: backend/src/main/java/com/dormpower/cache/lock/DistributedLockService.java

package com.dormpower.cache.lock;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.data.redis.core.script.DefaultRedisScript;
import org.springframework.stereotype.Service;

import java.time.Duration;
import java.util.Collections;
import java.util.UUID;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.TimeUnit;

/**
 * Redis分布式锁实现
 * 
 * 核心要求:
 * 1. 互斥性：同一时刻只有一个客户端能获取锁
 * 2. 不会死锁：锁有过期时间
 * 3. 可重入：同一线程可以多次获取
 * 4. 安全性：只能释放自己的锁
 * 
 * 实现方案:
 * - 使用SET NX EX命令（原子操作）
 * - Lua脚本保证释放锁的原子性
 * - WatchDog机制实现自动续期
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class DistributedLockService {

    private final RedisTemplate<String, Object> redisTemplate;

    // Lua脚本：安全释放锁（检查+删除原子操作）
    private static final String UNLOCK_SCRIPT = 
        "if redis.call('get', KEYS[1]) == ARGV[1] then " +
        "    return redis.call('del', KEYS[1]) " +
        "else " +
        "    return 0 " +
        "end";

    /**
     * 尝试获取锁
     * 
     * @param lockKey 锁Key
     * @param timeout 等待超时时间
     * @param leaseTime 锁持有时间
     * @return 锁标识（用于释放）
     */
    public String tryLock(String lockKey, Duration timeout, Duration leaseTime) {
        String lockValue = UUID.randomUUID().toString();
        long endTime = System.currentTimeMillis() + timeout.toMillis();

        while (System.currentTimeMillis() < endTime) {
            // SET NX EX 原子操作
            Boolean success = redisTemplate.opsForValue()
                .setIfAbsent(lockKey, lockValue, leaseTime);

            if (Boolean.TRUE.equals(success)) {
                log.debug("Lock acquired: {} with value: {}", lockKey, lockValue);
                return lockValue;
            }

            // 等待100ms后重试
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                return null;
            }
        }

        log.warn("Lock timeout: {}", lockKey);
        return null;
    }

    /**
     * 释放锁（使用Lua脚本保证原子性）
     */
    public boolean unlock(String lockKey, String lockValue) {
        DefaultRedisScript<Long> script = new DefaultRedisScript<>(UNLOCK_SCRIPT, Long.class);

        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(lockKey),
            lockValue
        );

        boolean success = result != null && result == 1;

        if (success) {
            log.debug("Lock released: {}", lockKey);
        } else {
            log.warn("Lock release failed (may be expired): {}", lockKey);
        }

        return success;
    }

    /**
     * 带锁的执行方法
     */
    public <T> T executeWithLock(String lockKey, Duration timeout, Duration leaseTime, LockSupplier<T> supplier) {
        String lockValue = tryLock(lockKey, timeout, leaseTime);

        if (lockValue == null) {
            throw new RuntimeException("Failed to acquire lock: " + lockKey);
        }

        try {
            return supplier.get();
        } finally {
            unlock(lockKey, lockValue);
        }
    }

    @FunctionalInterface
    public interface LockSupplier<T> {
        T get() throws Exception;
    }
}

2.5 CacheProtectionService 缓存保护

文件位置: backend/src/main/java/com/dormpower/cache/protection/CacheProtectionService.java

package com.dormpower.cache.protection;

import com.dormpower.cache.lock.DistributedLockService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.time.Duration;
import java.util.concurrent.Callable;
import java.util.concurrent.ThreadLocalRandom;

/**
 * 缓存保护服务
 * 
 * 解决三大缓存问题:
 * 
 * 1. 缓存穿透:
 *    - 问题: 查询不存在的数据，缓存不命中，每次打到数据库
 *    - 解决: 布隆过滤器 + 空值缓存
 * 
 * 2. 缓存击穿:
 *    - 问题: 热点Key过期瞬间，大量请求打到数据库
 *    - 解决: 分布式锁 + 互斥重建
 * 
 * 3. 缓存雪崩:
 *    - 问题: 大量Key同时过期，数据库压力骤增
 *    - 解决: 随机过期时间
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class CacheProtectionService {

    private final DistributedLockService distributedLock;

    /**
     * 带保护的缓存执行
     * 
     * 使用分布式锁防止缓存击穿
     */
    public <T> T executeWithProtection(String key, Callable<T> loader) {
        String lockKey = "lock:cache:" + key;

        try {
            // 尝试获取锁（等待3秒，持有10秒）
            return distributedLock.executeWithLock(
                lockKey,
                Duration.ofSeconds(3),
                Duration.ofSeconds(10),
                () -> loader.call()
            );
        } catch (Exception e) {
            log.error("Cache protection execution failed: {}", key, e);
            try {
                return loader.call();  // 降级：直接加载
            } catch (Exception ex) {
                throw new RuntimeException(ex);
            }
        }
    }

    /**
     * 添加随机过期时间（防雪崩）
     * 
     * @param baseTtl 基础过期时间
     * @return 添加抖动后的过期时间
     */
    public Duration addJitter(Duration baseTtl) {
        // 添加 ±10% 的随机抖动
        long jitter = ThreadLocalRandom.current()
            .nextLong((long) (baseTtl.toMillis() * 0.9), (long) (baseTtl.toMillis() * 1.1));

        return Duration.ofMillis(jitter);
    }

    /**
     * 缓存空值（防穿透）
     * 
     * 当数据库查询为空时，缓存一个空值标记
     */
    public void cacheNullValue(String key, Duration ttl) {
        // 缓存空对象，设置较短的过期时间
        String nullMarker = "NULL_MARKER";
        redisTemplate.opsForValue().set(key, nullMarker, Duration.ofMinutes(5));
    }
}

3.1 MultiLevelCacheConfig 配置

文件位置: backend/src/main/java/com/dormpower/config/MultiLevelCacheConfig.java

package com.dormpower.config;

import com.github.benmanes.caffeine.cache.Caffeine;
import org.springframework.cache.CacheManager;
import org.springframework.cache.caffeine.CaffeineCacheManager;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.cache.RedisCacheConfiguration;
import org.springframework.data.redis.cache.RedisCacheManager;
import org.springframework.data.redis.connection.RedisConnectionFactory;

import java.time.Duration;
import java.util.concurrent.TimeUnit;

/**
 * 多级缓存配置
 * 
 * L1 Caffeine配置:
 * - 最大10000条记录
 * - 写入后10分钟过期
 * - 记录统计信息
 * 
 * L2 Redis配置:
 * - 默认30分钟过期
 * - 使用前缀"cache:"
 * - 禁用缓存null值
 */
@Configuration
public class MultiLevelCacheConfig {

    @Bean
    public Caffeine<Object, Object> caffeineConfig() {
        return Caffeine.newBuilder()
            .maximumSize(10000)                    // 最大10000条
            .expireAfterWrite(10, TimeUnit.MINUTES) // 10分钟过期
            .recordStats()                         // 统计信息
            .removalListener((key, value, cause) -> 
                // 缓存移除通知
                log.debug("Cache removed: {}, reason: {}", key, cause));
    }

    @Bean
    public RedisCacheConfiguration redisCacheConfiguration() {
        return RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofMinutes(30))      // 30分钟过期
            .prefixCacheNameWith("cache:")         // 前缀
            .disableCachingNullValues();           // 不缓存null
    }
}

四、实战场景

4.1 设备状态缓存

@Service
@RequiredArgsConstructor
public class DeviceCacheService {

    private final MultiLevelCache multiLevelCache;
    private final DeviceRepository deviceRepository;

    /**
     * 获取设备信息（带多级缓存）
     */
    public Device getDevice(String deviceId) {
        String cacheKey = "device:" + deviceId;

        return multiLevelCache.get(
            cacheKey,
            () -> deviceRepository.findById(deviceId)
                .orElse(null),
            Duration.ofMinutes(30)
        );
    }

    /**
     * 更新设备信息（双删策略）
     */
    public Device updateDevice(Device device) {
        String cacheKey = "device:" + device.getId();

        // 1. 先删除缓存
        multiLevelCache.delete(cacheKey);

        // 2. 更新数据库
        Device updated = deviceRepository.save(device);

        // 3. 延迟删除在MultiLevelCache内部处理

        return updated;
    }
}

4.2 缓存监控指标

文件位置: backend/src/main/java/com/dormpower/cache/metrics/CacheMetricsService.java

package com.dormpower.cache.metrics;

import com.github.benmanes.caffeine.cache.Cache;
import io.micrometer.core.instrument.Gauge;
import io.micrometer.core.instrument.MeterRegistry;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import jakarta.annotation.PostConstruct;

/**
 * 缓存监控指标服务
 * 
 * 监控指标:
 * - 命中率（Hit Rate）
 * - 缓存大小（Size）
 * - 加载时间（Load Time）
 * - 驱逐次数（Eviction Count）
 */
@Service
@RequiredArgsConstructor
public class CacheMetricsService {

    private final Cache<String, Object> localCache;
    private final MeterRegistry meterRegistry;

    @PostConstruct
    public void initMetrics() {
        // L1缓存命中率
        Gauge.builder("cache.l1.hit.rate", localCache, 
            cache -> cache.stats().hitRate())
            .tag("cache", "caffeine")
            .register(meterRegistry);

        // L1缓存大小
        Gauge.builder("cache.l1.size", localCache, Cache::estimatedSize)
            .tag("cache", "caffeine")
            .register(meterRegistry);

        // L1缓存驱逐次数
        Gauge.builder("cache.l1.eviction.count", localCache, 
            cache -> cache.stats().evictionCount())
            .tag("cache", "caffeine")
            .register(meterRegistry);
    }
}
