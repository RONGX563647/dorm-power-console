# 告警管理模块 - 功能设计与实现细节文档

## 1. 模块概述

### 1.1 模块定位

告警管理模块是宿舍电源管理系统的安全保障模块，负责实现设备告警的配置、触发、通知和处理闭环。该模块提供告警规则配置、告警触发、告警通知、告警确认和告警历史查询等功能，确保系统异常能够及时发现和处理。

#### 2.1.6 实现代码

```java
@Service
public class AlertService {

    @Autowired
    private DeviceAlertConfigRepository deviceAlertConfigRepository;

    @Autowired
    private DeviceRepository deviceRepository;

    /**
     * 获取设备告警配置
     */
    @Cacheable(value = "alertConfigs", key = "#deviceId")
    public List<DeviceAlertConfig> getAlertConfigs(String deviceId) {
        return deviceAlertConfigRepository.findByDeviceId(deviceId);
    }

    /**
     * 获取设备指定类型的告警配置
     */
    @Cacheable(value = "alertConfigs", key = "#deviceId + '_' + #type")
    public DeviceAlertConfig getAlertConfig(String deviceId, String type) {
        return deviceAlertConfigRepository.findByDeviceIdAndType(deviceId, type)
            .orElseGet(() -> createDefaultConfig(deviceId, type));
    }

    /**
     * 更新告警配置
     */
    @CacheEvict(value = "alertConfigs", allEntries = true)
    public DeviceAlertConfig updateAlertConfig(String deviceId, String type, 
            double thresholdMin, double thresholdMax, boolean enabled) {
        
        Device device = deviceRepository.findById(deviceId)
            .orElseThrow(() -> new RuntimeException("Device not found: " + deviceId));

        DeviceAlertConfig config = deviceAlertConfigRepository
            .findByDeviceIdAndType(deviceId, type)
            .orElse(new DeviceAlertConfig());

        if (config.getId() == null) {
            config.setId("config_" + UUID.randomUUID().toString().substring(0, 8));
            config.setDeviceId(deviceId);
            config.setType(type);
            config.setCreatedAt(System.currentTimeMillis() / 1000);
        }

        config.setThresholdMin(thresholdMin);
        config.setThresholdMax(thresholdMax);
        config.setEnabled(enabled);
        config.setUpdatedAt(System.currentTimeMillis() / 1000);

        return deviceAlertConfigRepository.save(config);
    }

    /**
     * 创建默认告警配置
     */
    private DeviceAlertConfig createDefaultConfig(String deviceId, String type) {
        DeviceAlertConfig config = new DeviceAlertConfig();
        config.setId("config_" + UUID.randomUUID().toString().substring(0, 8));
        config.setDeviceId(deviceId);
        config.setType(type);
        config.setEnabled(true);
        config.setCreatedAt(System.currentTimeMillis() / 1000);
        config.setUpdatedAt(System.currentTimeMillis() / 1000);

        switch (type) {
            case "power":
                config.setThresholdMin(0);
                config.setThresholdMax(200);
                break;
            case "voltage":
                config.setThresholdMin(200);
                config.setThresholdMax(240);
                break;
            case "current":
                config.setThresholdMin(0);
                config.setThresholdMax(10);
                break;
            default:
                config.setThresholdMin(0);
                config.setThresholdMax(100);
        }

        return deviceAlertConfigRepository.save(config);
    }
}
```

#### 2.2.5 实现代码

```java
@Service
public class AlertService {

    @Autowired
    private DeviceAlertRepository deviceAlertRepository;

    @Autowired
    private DeviceAlertConfigRepository deviceAlertConfigRepository;

    @Autowired
    private WebSocketNotificationService webSocketNotificationService;

    /**
     * 检查并生成告警
     */
    public DeviceAlert checkAndGenerateAlert(String deviceId, String type, double value) {
        DeviceAlertConfig config = getAlertConfig(deviceId, type);
        
        if (!config.isEnabled()) {
            return null;
        }

        // 检查是否存在未解决的同类告警
        List<DeviceAlert> unresolvedAlerts = deviceAlertRepository
            .findByDeviceIdAndTypeAndResolvedFalse(deviceId, type);
        if (!unresolvedAlerts.isEmpty()) {
            return null;
        }

        long now = System.currentTimeMillis() / 1000;
        String level = "info";
        String message = "";

        if (type.equals("power")) {
            if (value > config.getThresholdMax()) {
                level = "warning";
                message = String.format("功率超过阈值: %.1fW (最大: %.1fW)", 
                    value, config.getThresholdMax());
            } else if (value < config.getThresholdMin()) {
                level = "info";
                message = String.format("功率低于阈值: %.1fW (最小: %.1fW)", 
                    value, config.getThresholdMin());
            }
        } else if (type.equals("voltage")) {
            if (value > config.getThresholdMax()) {
                level = "error";
                message = String.format("电压超过阈值: %.1fV (最大: %.1fV)", 
                    value, config.getThresholdMax());
            } else if (value < config.getThresholdMin()) {
                level = "error";
                message = String.format("电压低于阈值: %.1fV (最小: %.1fV)", 
                    value, config.getThresholdMin());
            }
        } else if (type.equals("current")) {
            if (value > config.getThresholdMax()) {
                level = "warning";
                message = String.format("电流超过阈值: %.1fA (最大: %.1fA)", 
                    value, config.getThresholdMax());
            }
        }

        if (!message.isEmpty()) {
            DeviceAlert alert = new DeviceAlert();
            alert.setId("alert_" + System.currentTimeMillis() + "_" + 
                (int)(Math.random() * 1000));
            alert.setDeviceId(deviceId);
            alert.setType(type);
            alert.setLevel(level);
            alert.setMessage(message);
            alert.setThresholdValue(type.equals("power") || type.equals("current") ? 
                config.getThresholdMax() : 
                (value > config.getThresholdMax() ? config.getThresholdMax() : config.getThresholdMin()));
            alert.setActualValue(value);
            alert.setResolved(false);
            alert.setTs(now);
            alert.setCreatedAt(now);

            DeviceAlert savedAlert = deviceAlertRepository.save(alert);

            // WebSocket推送告警
            webSocketNotificationService.broadcastAlert(deviceId, savedAlert);

            return savedAlert;
        }

        return null;
    }
}
```

#### 2.3.4 实现代码

```java
@Service
public class AlertNotificationService {

    @Autowired
    private WebSocketNotificationService webSocketNotificationService;

    @Autowired
    private EmailService emailService;

    /**
     * 发送告警通知
     */
    public void sendAlertNotification(DeviceAlert alert, Device device) {
        // WebSocket实时推送
        sendWebSocketNotification(alert, device);

        // 邮件通知（error及以上级别）
        if ("error".equals(alert.getLevel()) || "critical".equals(alert.getLevel())) {
            sendEmailNotification(alert, device);
        }
    }

    /**
     * WebSocket通知
     */
    private void sendWebSocketNotification(DeviceAlert alert, Device device) {
        Map<String, Object> notification = new HashMap<>();
        notification.put("alertId", alert.getId());
        notification.put("deviceId", alert.getDeviceId());
        notification.put("deviceName", device.getName());
        notification.put("room", device.getRoom());
        notification.put("type", alert.getType());
        notification.put("level", alert.getLevel());
        notification.put("message", alert.getMessage());
        notification.put("actualValue", alert.getActualValue());
        notification.put("thresholdValue", alert.getThresholdValue());
        notification.put("timestamp", alert.getTs());

        webSocketNotificationService.broadcastAlert(alert.getDeviceId(), notification);
    }

    /**
     * 邮件通知
     */
    private void sendEmailNotification(DeviceAlert alert, Device device) {
        String subject = String.format("[%s] 设备告警 - %s", 
            alert.getLevel().toUpperCase(), device.getName());
        
        String content = String.format(
            "设备名称: %s\n" +
            "所在房间: %s\n" +
            "告警类型: %s\n" +
            "告警级别: %s\n" +
            "告警消息: %s\n" +
            "实际值: %.2f\n" +
            "阈值: %.2f\n" +
            "告警时间: %s\n",
            device.getName(),
            device.getRoom(),
            alert.getType(),
            alert.getLevel(),
            alert.getMessage(),
            alert.getActualValue(),
            alert.getThresholdValue(),
            Instant.ofEpochSecond(alert.getTs()).toString()
        );

        // 发送给管理员
        emailService.sendToAdmins(subject, content);
    }
}
```

```java
@RestController
@RequestMapping("/api/alerts")
public class AlertController {

    @Autowired
    private AlertService alertService;

    /**
     * 解决告警
     */
    @Operation(summary = "解决告警", description = "标记告警为已解决状态")
    @AuditLog(value = "解决告警", type = "ALERT")
    @PostMapping("/{alertId}/resolve")
    public ResponseEntity<?> resolveAlert(
            @PathVariable String alertId,
            @RequestHeader("Authorization") String authHeader) {
        
        String username = jwtUtil.getUsernameFromToken(authHeader.replace("Bearer ", ""));
        
        DeviceAlert alert = alertService.resolveAlert(alertId, username);
        
        return ResponseEntity.ok(alert);
    }
}
```

```java
@Service
public class AlertService {

    /**
     * 解决告警
     */
    public DeviceAlert resolveAlert(String alertId, String resolvedBy) {
        DeviceAlert alert = deviceAlertRepository.findById(alertId)
            .orElseThrow(() -> new RuntimeException("Alert not found: " + alertId));

        alert.setResolved(true);
        alert.setResolvedAt(System.currentTimeMillis() / 1000);
        alert.setResolvedBy(resolvedBy);

        return deviceAlertRepository.save(alert);
    }
}
```

```java
@Service
public class AlertService {

    /**
     * 获取设备告警列表
     */
    public List<DeviceAlert> getAlertsByDeviceId(String deviceId) {
        return deviceAlertRepository.findByDeviceIdOrderByTsDesc(deviceId);
    }

    /**
     * 获取未解决告警列表
     */
    public List<DeviceAlert> getUnresolvedAlerts() {
        return deviceAlertRepository.findByResolvedFalseOrderByTsDesc();
    }

    /**
     * 获取所有告警
     */
    public List<DeviceAlert> getAllAlerts() {
        return deviceAlertRepository.findAll();
    }

    /**
     * 获取时间范围内的告警
     */
    public List<DeviceAlert> getAlertsByTimeRange(long startTs, long endTs) {
        return deviceAlertRepository.findByTsBetweenOrderByTsDesc(startTs, endTs);
    }
}
```
