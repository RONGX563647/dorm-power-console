```java
@Component
public class PerformanceMonitor {
    
    private final MeterRegistry meterRegistry;
    private final Timer requestTimer;
    
    public PerformanceMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        this.requestTimer = Timer.builder("http.request.duration")
            .tag("application", "dorm-power")
            .register(meterRegistry);
    }
    
    public <T> T monitor(Supplier<T> operation, String operationName) {
        return requestTimer.record(() -> {
            long startTime = System.currentTimeMillis();
            try {
                return operation.get();
            } finally {
                long duration = System.currentTimeMillis() - startTime;
                if (duration > 1000) {
                    log.warn("Slow operation detected: {} took {}ms", 
                        operationName, duration);
                }
            }
        });
    }
}
```

#### 3.2.3 批量操作优化

```java
@Service
@RequiredArgsConstructor
public class BatchOptimizationService {
    
    private final JdbcTemplate jdbcTemplate;
    
    // 批量插入优化
    @Transactional
    public void batchInsertTelemetry(List<TelemetryData> dataList) {
        String sql = "INSERT INTO telemetry (device_id, voltage, current, power, collected_at) " +
                    "VALUES (?, ?, ?, ?, ?)";
        
        jdbcTemplate.batchUpdate(sql, new BatchPreparedStatementSetter() {
            @Override
            public void setValues(PreparedStatement ps, int i) throws SQLException {
                TelemetryData data = dataList.get(i);
                ps.setString(1, data.getDeviceId());
                ps.setDouble(2, data.getVoltage());
                ps.setDouble(3, data.getCurrent());
                ps.setDouble(4, data.getPower());
                ps.setTimestamp(5, Timestamp.valueOf(data.getCollectedAt()));
            }
            
            @Override
            public int getBatchSize() {
                return dataList.size();
            }
        });
    }
    
    // 分批处理（避免内存溢出）
    public void processLargeDataset(int totalRecords, int batchSize, Consumer<List<Integer>> processor) {
        int offset = 0;
        while (offset < totalRecords) {
            List<Integer> batch = IntStream.range(offset, Math.min(offset + batchSize, totalRecords))
                .boxed()
                .collect(Collectors.toList());
            
            processor.accept(batch);
            offset += batchSize;
        }
    }
}
```

```java
@Service
public class ObjectPoolService {
    
    // 使用对象池复用DateFormatter（线程不安全）
    private final ObjectPool<SimpleDateFormat> dateFormatterPool = new GenericObjectPool<>(
        new BasePooledObjectFactory<SimpleDateFormat>() {
            @Override
            public SimpleDateFormat create() {
                return new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
            }
            
            @Override
            public PooledObject<SimpleDateFormat> wrap(SimpleDateFormat obj) {
                return new DefaultPooledObject<>(obj);
            }
        }
    );
    
    public String formatDate(Date date) throws Exception {
        SimpleDateFormat sdf = null;
        try {
            sdf = dateFormatterPool.borrowObject();
            return sdf.format(date);
        } finally {
            if (sdf != null) {
                dateFormatterPool.returnObject(sdf);
            }
        }
    }
}
```

```java
    public synchronized void processDevicesBad(List<Device> devices) {
        // 整个方法加锁，串行执行
```

```java
// 改进: 多级缓存策略
@Service
public class MultiLevelCacheService {
    
    private final Cache<String, Device> localCache;  // Caffeine L1
    private final RedisTemplate<String, Device> redisTemplate;  // Redis L2
    
    public Device getDevice(String deviceId) {
        // L1: 本地缓存
        Device device = localCache.getIfPresent(deviceId);
        if (device != null) {
            return device;
        }
        
        // L2: Redis缓存
        device = redisTemplate.opsForValue().get("device:" + deviceId);
        if (device != null) {
            localCache.put(deviceId, device);  // 回填L1
            return device;
        }
        
        // L3: 数据库
        device = deviceRepository.findById(deviceId).orElse(null);
        if (device != null) {
            redisTemplate.opsForValue().set("device:" + deviceId, device, 
                Duration.ofMinutes(30));
            localCache.put(deviceId, device);
        }
        
        return device;
    }
    
    // 缓存更新策略: Cache-Aside
    public void updateDevice(Device device) {
        deviceRepository.save(device);
        localCache.invalidate(device.getDeviceId());
        redisTemplate.delete("device:" + device.getDeviceId());
    }
}
```

```java
        targetDataSources.put("read", readDataSource());
        
        RoutingDataSource routingDataSource = new RoutingDataSource();
        routingDataSource.setDefaultTargetDataSource(writeDataSource());
        routingDataSource.setTargetDataSources(targetDataSources);
        return routingDataSource;
    }
}
```

```python
class AnomalyDetector:
    """基于机器学习的异常检测"""
    
    def __init__(self):
        self.model = self.load_model()
    
    def detect_anomaly(self, telemetry_data: List[TelemetryData]) -> AnomalyResult:
        """检测用电异常"""
        # 特征工程
        features = self.extract_features(telemetry_data)
        
        # 模型预测
        prediction = self.model.predict(features)
        
        # 生成报告
        return AnomalyResult(
            is_anomaly=prediction['is_anomaly'],
            confidence=prediction['confidence'],
            anomaly_type=prediction['type'],
            suggestions=self.generate_suggestions(prediction)
        )
    
    def extract_features(self, data: List[TelemetryData]) -> np.ndarray:
        """提取特征: 电压波动、电流峰值、功率趋势等"""
        voltages = [d.voltage for d in data]
        currents = [d.current for d in data]
        powers = [d.power for d in data]
        
        return np.array([
            np.std(voltages),  # 电压标准差
            np.max(currents),  # 电流峰值
            np.polyfit(range(len(powers)), powers, 1)[0],  # 功率趋势
            # 更多特征...
        ])
```
