## 一、系统架构

### 1.1 为什么需要限流和熔断？

```yaml
系统保护机制:
  限流（Rate Limiting）:
    目的: 防止系统被过量请求打垮
    场景:
      - 恶意刷接口
      - 爬虫请求
      - 突发流量
  
  熔断（Circuit Breaker）:
    目的: 防止级联故障
    场景:
      - 下游服务不可用
      - 数据库连接耗尽
      - 外部API超时
  
  降级（Fallback）:
    目的: 保证核心功能可用
    场景:
      - 非核心功能失败
      - 缓存替代数据库
      - 默认值替代计算
```

### 2.1 RedisRateLimiter 分布式限流

**文件位置**: `backend/src/main/java/com/dormpower/limiter/RedisRateLimiter.java`

```java
package com.dormpower.limiter;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.data.redis.core.script.DefaultRedisScript;
import org.springframework.stereotype.Service;

import java.time.Instant;
import java.util.Collections;
import java.util.List;

/**
 * Redis分布式限流器
 * 
 * 算法: 滑动窗口
 * 
 * 原理:
 * 1. 使用Redis ZSet存储请求时间戳
 * 2. 查询窗口内的请求数
 * 3. 超过阈值则拒绝
 * 4. 自动清理过期数据
 * 
 * 优势:
 * - 分布式环境可用
 * - 精确控制
 * - 自动过期
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class RedisRateLimiter {

    private final RedisTemplate<String, Object> redisTemplate;
    
    // Lua脚本：原子操作
    private static final String RATE_LIMIT_SCRIPT = 
        "local key = KEYS[1] " +
        "local window = tonumber(ARGV[1]) " +
        "local limit = tonumber(ARGV[2]) " +
        "local now = tonumber(ARGV[3]) " +
        "local windowStart = now - window " +
        // 清理过期数据
        "redis.call('ZREMRANGEBYSCORE', key, 0, windowStart) " +
        // 获取当前窗口内的请求数
        "local count = redis.call('ZCARD', key) " +
        // 判断是否超过限制
        "if count >= limit then " +
        "    return 0 " +
        "end " +
        // 添加当前请求
        "redis.call('ZADD', key, now, now .. '-' .. math.random(1000000)) " +
        // 设置过期时间
        "redis.call('EXPIRE', key, window) " +
        "return 1";
    
    /**
     * 检查是否允许请求通过
     * 
     * @param key 限流键（如：IP、用户ID、接口）
     * @param windowSeconds 窗口大小（秒）
     * @param maxRequests 最大请求数
     * @return true=允许，false=拒绝
     */
    public boolean tryAcquire(String key, int windowSeconds, int maxRequests) {
        String redisKey = "rate_limit:" + key;
        long now = Instant.now().getEpochSecond();
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>(RATE_LIMIT_SCRIPT, Long.class);
        
        Long result = redisTemplate.execute(
            script,
            Collections.singletonList(redisKey),
            windowSeconds,
            maxRequests,
            now
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 获取当前窗口内的请求数
     */
    public long getCurrentCount(String key, int windowSeconds) {
        String redisKey = "rate_limit:" + key;
        long now = Instant.now().getEpochSecond();
        long windowStart = now - windowSeconds;
        
        // 清理过期数据
        redisTemplate.opsForZSet().removeRangeByScore(redisKey, 0, windowStart);
        
        // 获取当前窗口内的请求数
        Long count = redisTemplate.opsForZSet().zCard(redisKey);
        
        return count != null ? count : 0;
    }
}
```

### 2.2 ResilientCacheService 弹性缓存服务

**文件位置**: `backend/src/main/java/com/dormpower/cache/resilience/ResilientCacheService.java`

```java
package com.dormpower.cache.resilience;

import io.github.resilience4j.circuitbreaker.CircuitBreaker;
import io.github.resilience4j.circuitbreaker.CircuitBreakerConfig;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;

import java.time.Duration;
import java.util.concurrent.Callable;
import java.util.function.Supplier;

/**
 * 弹性缓存服务
 * 
 * 功能:
 * 1. 熔断器保护（Circuit Breaker）
 * 2. 失败降级（Fallback）
 * 3. 重试机制（Retry）
 * 
 * @author dormpower team
 * @version 1.0
 */
@Slf4j
@Service
public class ResilientCacheService {

    private final RedisTemplate<String, Object> redisTemplate;
    
    // 熔断器
    private final CircuitBreaker circuitBreaker;
    
    public ResilientCacheService(RedisTemplate<String, Object> redisTemplate) {
        this.redisTemplate = redisTemplate;
        
        // 配置熔断器
        CircuitBreakerConfig config = CircuitBreakerConfig.custom()
            .failureRateThreshold(50)           // 失败率>50%时熔断
            .waitDurationInOpenState(Duration.ofSeconds(30))  // 熔断30秒后尝试半开
            .slidingWindowSize(10)              // 滑动窗口大小
            .minimumNumberOfCalls(5)            // 最小调用次数
            .build();
        
        this.circuitBreaker = CircuitBreaker.of("redis", config);
    }
    
    /**
     * 带熔断保护的缓存查询
     */
    @SuppressWarnings("unchecked")
    public <T> T get(String key, Supplier<T> fallback) {
        try {
            return circuitBreaker.executeSupplier(() -> 
                (T) redisTemplate.opsForValue().get(key)
            );
        } catch (Exception e) {
            log.warn("Cache query failed, using fallback: {}", e.getMessage());
            return fallback.get();
        }
    }
    
    /**
     * 带熔断保护的缓存写入
     */
    public void put(String key, Object value, Duration ttl, Runnable fallback) {
        try {
            circuitBreaker.run(() -> {
                redisTemplate.opsForValue().set(key, value, ttl);
                return null;
            });
        } catch (Exception e) {
            log.warn("Cache write failed, executing fallback: {}", e.getMessage());
            fallback.run();
        }
    }
    
    /**
     * 获取熔断器状态
     */
    public String getCircuitBreakerState() {
        return circuitBreaker.getState().toString();
    }
}
```

## 三、应用场景

### 3.1 接口限流

```java
@RestController
@RequestMapping("/api/devices")
@RequiredArgsConstructor
public class DeviceController {

    private final RedisRateLimiter rateLimiter;
    
    @GetMapping("/{id}")
    public ResponseEntity<Device> getDevice(@PathVariable String id) {
        // 限流：每IP每分钟最多100次
        String key = "device_query:" + getClientIp();
        if (!rateLimiter.tryAcquire(key, 60, 100)) {
            return ResponseEntity.status(429).build();  // Too Many Requests
        }
        
        Device device = deviceService.getDevice(id);
        return ResponseEntity.ok(device);
    }
}
```
